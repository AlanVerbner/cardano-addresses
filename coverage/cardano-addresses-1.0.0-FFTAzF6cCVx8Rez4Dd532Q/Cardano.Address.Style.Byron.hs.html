   85 -- $overview
   86 --
   87 -- This module provides an implementation of:
   88 --
   89 -- - 'GenMasterKey': for generating Byron master keys from mnemonic sentences
   90 -- - 'HardDerivation': for hierarchical derivation of parent to child keys
   91 -- - 'PaymentAddress': for constructing addresses from a public key
   92 --
   93 -- We call 'Byron' addresses the old address type used by Daedalus in the early
   94 -- days of Cardano. Using this type of addresses and underlying key scheme is
   95 -- now considered __deprecated__ because of some security implications.
   96 --
   97 -- Unless you have good reason to do so (like writing backward-compatible code
   98 -- with an existing piece), any new implementation __should use__ the 'Icarus'
   99 -- style for key and addresses.
  100 --
  101 -- The internals of the 'Byron' does not matter for the reader, but basically
  102 -- contains what is necessary to perform key derivation and generate addresses
  103 -- from a 'Byron' type.
  104

  109 -- | Material for deriving HD random scheme keys, which can be used for making
  110 -- addresses.
  111 --
  112 -- @since 1.0.0
  113 data Byron (depth :: Depth) key = Byron
  114     { getKey :: key
  115     -- ^ The raw private or public key.
  116     --
  117     -- @since 1.0.0
  118     , derivationPath :: DerivationPath depth
  119     -- ^ The address derivation indices for the level of this key.
  120     --
  121     -- @since 1.0.0
  122     , payloadPassphrase :: ScrubbedBytes
  123     -- ^ Used for encryption of the derivation path payload within an address.
  124     --
  125     -- @since 1.0.0
  126     } deriving stock (Generic)
  127 {-# DEPRECATED Byron "see Cardano.Address.Style.Icarus" #-}
  128 {-# DEPRECATED getKey "see Cardano.Address.Style.Icarus" #-}
  129 {-# DEPRECATED derivationPath "see Cardano.Address.Style.Icarus" #-}
  130 {-# DEPRECATED payloadPassphrase "see Cardano.Address.Style.Icarus" #-}
  131 
  132 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
  133 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
  134 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
  135 deriving instance (Functor (Byron depth))

  137 -- | Backdoor for generating a new key from a raw XPrv.
  138 --
  139 -- Note that the @depth@ is left open so that the caller gets to decide what type
  140 -- of key this is. This is mostly for testing, in practice, seeds are used to
  141 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  142 --
  143 -- The first argument is a type-family 'DerivationPath' and its type depends on
  144 -- the 'depth' of the key.
  145 --
  146 -- __examples:__
  147 --
  148 -- >>> liftXPrv () prv
  149 -- _ :: Byron RootK XPrv
  150 --
  151 -- >>> liftXPrv minBound prv
  152 -- _ :: Byron AccountK XPrv
  153 --
  154 -- >>> liftXPrv (minBound, minBound) prv
  155 -- _ :: Byron AddressK XPrv
  156 --
  157 -- @since 1.0.0
  158 liftXPrv
  159     :: DerivationPath depth
  160     -> XPrv
  161     -> Byron depth XPrv
  162 liftXPrv derivationPath getKey = Byron
  163     { getKey
  164     , derivationPath
  165     , payloadPassphrase = hdPassphrase (toXPub getKey)
  166     }
  167 {-# DEPRECATED liftXPrv "see Cardano.Address.Style.Icarus" #-}
  168 
  169 -- | The hierarchical derivation indices for a given level/depth.
  170 --
  171 -- @since 1.0.0
  172 type family DerivationPath (depth :: Depth) :: * where
  173     -- The root key is generated from the seed.
  174     DerivationPath 'RootK =
  175         ()
  176     -- The account key is generated from the root key and account index.
  177     DerivationPath 'AccountK =
  178         Index 'WholeDomain 'AccountK
  179     -- The address key is generated from the account key and address index.
  180     DerivationPath 'AddressK =
  181         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  182 {-# DEPRECATED DerivationPath "see Cardano.Address.Style.Icarus" #-}
  183 
  184 instance GenMasterKey Byron where
  185     type SecondFactor Byron = ()
  186 
  187     genMasterKeyFromXPrv = liftXPrv ()
  188     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  189         liftXPrv () xprv
  190       where
  191         xprv = generate (hashSeed seedValidated)
  192         seed  = entropyToBytes $ mnemonicToEntropy mw
  193         seedValidated = assert
  194             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  195             seed
  196 
  197 instance HardDerivation Byron where
  198     type AddressIndexDerivationType Byron = 'WholeDomain
  199     type AccountIndexDerivationType Byron = 'WholeDomain
  200     type WithAccountStyle Byron = ()
  201 
  202     deriveAccountPrivateKey rootXPrv accIx = Byron
  203         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  204         , derivationPath = accIx
  205         , payloadPassphrase = payloadPassphrase rootXPrv
  206         }
  207 
  208     deriveAddressPrivateKey accXPrv () addrIx = Byron
  209         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  210         , derivationPath = (derivationPath accXPrv, addrIx)
  211         , payloadPassphrase = payloadPassphrase accXPrv
  212         }

  214 instance PaymentAddress Byron where
  215     paymentAddress discrimination k = unsafeMkAddress
  216         $ CBOR.toStrictByteString
  217         $ CBOR.encodeAddress (getKey k) attrs
  218       where
  219         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  220         pwd = payloadPassphrase k
  221         attrs = case discrimination of
  222             RequiresNoMagic ->
  223                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  224                 ]
  225             RequiresMagic (ProtocolMagic pm) ->
  226                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  227                 , CBOR.encodeProtocolMagicAttr pm
  228                 ]

  234 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  235 minSeedLengthBytes :: Int
  236 minSeedLengthBytes = 16
  237 
  238 -- | Hash the seed entropy (generated from mnemonic) used to initiate a HD
  239 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  240 -- minimum for 'generate' (32 bytes).
  241 --
  242 -- Note that our current implementation deviates from BIP-39 because we use a
  243 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  244 --
  245 -- There are two methods of hashing the seed entropy, for different use cases.
  246 --
  247 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  248 --    Blake2b_256, inside a double CBOR serialization sandwich.
  249 --
  250 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  251 --    Blake2b_256, without any serialization.
  252 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  253 hashSeed = serialize . blake2b256 . serialize
  254   where
  255     serialize = BA.convert . cbor . BA.convert
  256     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  257 
  258 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  259 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  260 
  261 -- | Derive a symmetric key for encrypting and authenticating the address
  262 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  263 -- is employed.
  264 hdPassphrase :: XPub -> ScrubbedBytes
  265 hdPassphrase masterKey =
  266     PBKDF2.generate
  267     (PBKDF2.prfHMAC SHA512)
  268     (PBKDF2.Parameters 500 32)
  269     (xpubToBytes masterKey)
  270     ("address-hashing" :: ByteString)

  273 word32 = fromIntegral . fromEnum
