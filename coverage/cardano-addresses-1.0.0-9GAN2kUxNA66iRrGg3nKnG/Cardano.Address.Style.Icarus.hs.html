   92 -- | A cryptographic key for sequential-scheme address derivation, with
   93 -- phantom-types to disambiguate key types.
   94 --
   95 -- @
   96 -- let rootPrivateKey = Icarus 'RootK XPrv
   97 -- let accountPubKey = Icarus 'AccountK XPub
   98 -- let addressPubKey = Icarus 'AddressK XPub
   99 -- @
  100 newtype Icarus (depth :: Depth) key =
  101     Icarus { getKey :: key }
  102     deriving stock (Generic, Show, Eq)
  103 
  104 deriving instance (Functor (Icarus depth))
  105 instance (NFData key) => NFData (Icarus depth key)
  106 
  107 instance GenMasterKey Icarus where
  108     type GenMasterKeyFrom Icarus = SomeMnemonic
  109 
  110     genMasterKey = unsafeGenerateKeyFromSeed

  112 instance HardDerivation Icarus where
  113     type AccountIndexDerivationType Icarus = 'Hardened
  114     type AddressIndexDerivationType Icarus = 'Soft
  115 
  116     deriveAccountPrivateKey (Icarus rootXPrv) accIx =
  117         let
  118             purposeIx =
  119                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  120             coinTypeIx =
  121                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  122             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  123                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  124             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  125                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  126             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  127                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  128         in
  129             Icarus acctXPrv
  130 
  131     deriveAddressPrivateKey (Icarus accXPrv) accountingStyle addrIx =
  132         let
  133             changeCode =
  134                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  135             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  136                 deriveXPrv DerivationScheme2 accXPrv changeCode
  137             addrXPrv = -- lvl5 derivation; soft derivation of address index
  138                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  139         in
  140             Icarus addrXPrv
  141 
  142 instance SoftDerivation Icarus where
  143     deriveAddressPublicKey (Icarus accXPub) accountingStyle addrIx =
  144         fromMaybe errWrongIndex $ do
  145             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  146             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  147                 deriveXPub DerivationScheme2 accXPub changeCode
  148             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  149                 deriveXPub DerivationScheme2 changeXPub addrIx
  150             return $ Icarus addrXPub
  151       where
  152         errWrongIndex = error $
  153             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  154             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  155             \either a programmer error, or, we may have reached the maximum \
  156             \number of addresses for a given wallet."
  157 
  158 instance PaymentAddress Icarus where
  159     paymentAddress discrimination k = unsafeMkAddress
  160         $ CBOR.toStrictByteString

  173 -- | Purpose is a constant set to 44' (or 0x8000002C) following the original
  174 -- BIP-44 specification.
  175 --
  176 -- It indicates that the subtree of this node is used according to this
  177 -- specification.
  178 --
  179 -- Hardened derivation is used at this level.
  180 purposeIndex :: Word32
  181 purposeIndex = 0x8000002C
  182 
  183 -- | One master node (seed) can be used for unlimited number of independent
  184 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  185 -- same space for various cryptocoins has some disadvantages.
  186 --
  187 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  188 -- addresses across cryptocoins and improving privacy issues.
  189 --
  190 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  191 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  192 -- Lovelace.
  193 --
  194 -- Hardened derivation is used at this level.
  195 coinTypeIndex :: Word32
  196 coinTypeIndex = 0x80000717

  198 -- | The minimum seed length for 'generateKeyFromSeed' and 'unsafeGenerateKeyFromSeed'.
  199 minSeedLengthBytes :: Int
  200 minSeedLengthBytes = 16
  201 
  202 -- | Hardware Ledger devices generates keys from mnemonic using a different
  203 -- approach (different from the rest of Cardano).
  204 --
  205 -- It is a combination of:
  206 --
  207 -- - [SLIP 0010](https://github.com/satoshilabs/slips/blob/master/slip-0010.md)
  208 -- - [BIP 0032](https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki)
  209 -- - [BIP 0039](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki)
  210 -- - [RFC 8032](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  211 -- - What seems to be arbitrary changes from Ledger regarding the calculation of
  212 --   the initial chain code and generation of the root private key.
  213 unsafeGenerateKeyFromHardwareLedger
  214     :: SomeMnemonic
  215         -- ^ The root mnemonic
  216     -> Icarus 'RootK XPrv
  217 unsafeGenerateKeyFromHardwareLedger (SomeMnemonic mw) = unsafeFromRight $ do
  218     let seed = pbkdf2HmacSha512
  219             $ T.encodeUtf8
  220             $ T.intercalate " "
  221             $ mnemonicToText mw
  222 
  223     -- NOTE
  224     -- SLIP-0010 refers to `iR` as the chain code. Here however, the chain code
  225     -- is obtained as a hash of the initial seed whereas iR is used to make part
  226     -- of the root private key itself.
  227     let cc = hmacSha256 (BS.pack [1] <> seed)
  228     let (iL, iR) = first pruneBuffer $ hashRepeatedly seed
  229 
  230     prv <- maybe (Left "invalid xprv") pure $ xprvFromBytes $ iL <> iR <> cc
  231     pure $ Icarus prv
  232   where
  233     -- Errors yielded in the body of 'unsafeGenerateKeyFromHardwareLedger' are
  234     -- programmer errors (out-of-range byte buffer access or, invalid length for
  235     -- cryptographic operations). Therefore, we throw badly if we encounter any.
  236     unsafeFromRight :: Either String a -> a
  237     unsafeFromRight = either error id
  238 
  239     -- This is the algorithm described in SLIP 0010 for master key generation
  240     -- with an extra step to discard _some_ of the potential private keys. Why
  241     -- this extra step remains a mystery as of today.
  242     --
  243     --      1. Generate a seed byte sequence S of 512 bits according to BIP-0039.
  244     --         (done in a previous step, passed as argument).
  245     --
  246     --      2. Calculate I = HMAC-SHA512(Key = "ed25519 seed", Data = S)
  247     --
  248     --      3. Split I into two 32-byte sequences, IL and IR.
  249     --
  250     -- extra *******************************************************************
  251     -- *                                                                       *
  252     -- *    3.5 If the third highest bit of the last byte of IL is not zero    *
  253     -- *        S = I and go back to step 2.                                   *
  254     -- *                                                                       *
  255     -- *************************************************************************
  256     --
  257     --      4. Use parse256(IL) as master secret key, and IR as master chain code.
  258     hashRepeatedly :: ByteString -> (ByteString, ByteString)
  259     hashRepeatedly bytes = case BS.splitAt 32 (hmacSha512 bytes) of
  260         (iL, iR) | isInvalidKey iL -> hashRepeatedly (iL <> iR)
  261         (iL, iR) -> (iL, iR)
  262       where
  263         isInvalidKey k = testBit (k `BS.index` 31) 5
  264 
  265     -- - Clear the lowest 3 bits of the first byte
  266     -- - Clear the highest bit of the last byte
  267     -- - Set the second highest bit of the last byte
  268     --
  269     -- As described in [RFC 8032 - 5.1.5](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  270     pruneBuffer :: ByteString -> ByteString
  271     pruneBuffer bytes =
  272         let
  273             (firstByte, rest) = fromMaybe (error "pruneBuffer: no first byte") $
  274                 BS.uncons bytes
  275 
  276             (rest', lastByte) = fromMaybe (error "pruneBuffer: no last byte") $
  277                 BS.unsnoc rest
  278 
  279             firstPruned = firstByte
  280                 & (`clearBit` 0)
  281                 & (`clearBit` 1)
  282                 & (`clearBit` 2)
  283 
  284             lastPruned = lastByte
  285                 & (`setBit` 6)
  286                 & (`clearBit` 7)
  287         in
  288             (firstPruned `BS.cons` BS.snoc rest' lastPruned)
  289 
  290     -- As described in [BIP 0039 - From Mnemonic to Seed](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki#from-mnemonic-to-seed)
  291     pbkdf2HmacSha512 :: ByteString -> ByteString
  292     pbkdf2HmacSha512 bytes = PBKDF2.generate
  293         (PBKDF2.prfHMAC SHA512)
  294         (PBKDF2.Parameters 2048 64)
  295         bytes
  296         ("mnemonic" :: ByteString)
  297 
  298     hmacSha256 :: ByteString -> ByteString
  299     hmacSha256 =
  300         BA.convert @(HMAC SHA256) . hmac salt
  301 
  302     -- As described in [SLIP 0010 - Master Key Generation](https://github.com/satoshilabs/slips/blob/master/slip-0010.md#master-key-generation)
  303     hmacSha512 :: ByteString -> ByteString
  304     hmacSha512 =
  305         BA.convert @(HMAC SHA512) . hmac salt
  306 
  307     salt :: ByteString
  308     salt = "ed25519 seed"

  310 -- | Generate a new key from seed. Note that the @depth@ is left open so that
  311 -- the caller gets to decide what type of key this is. This is mostly for
  312 -- testing, in practice, seeds are used to represent root keys, and one should
  313 -- use 'generateKeyFromSeed'.
  314 unsafeGenerateKeyFromSeed
  315     :: SomeMnemonic
  316         -- ^ The root mnemonic
  317     -> Icarus depth XPrv
  318 unsafeGenerateKeyFromSeed (SomeMnemonic mw) =
  319     let
  320         seed  = entropyToBytes $ mnemonicToEntropy mw
  321         seedValidated = assert
  322             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  323             seed
  324     in Icarus $ generateNew seedValidated (mempty :: ScrubbedBytes)
