   93 -- | Material for deriving HD random scheme keys, which can be used for making
   94 -- addresses.
   95 data Byron (depth :: Depth) key = Byron
   96     { getKey :: key
   97     -- ^ The raw private or public key.
   98     , derivationPath :: DerivationPath depth
   99     -- ^ The address derivation indices for the level of this key.
  100     , payloadPassphrase :: ScrubbedBytes
  101     -- ^ Used for encryption of payload containing address derivation path.
  102     } deriving stock (Generic)
  103 
  104 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
  105 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
  106 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
  107 
  108 -- | The hierarchical derivation indices for a given level/depth.
  109 type family DerivationPath (depth :: Depth) :: * where
  110     -- The root key is generated from the seed.
  111     DerivationPath 'RootK =
  112         ()
  113     -- The account key is generated from the root key and account index.
  114     DerivationPath 'AccountK =
  115         Index 'WholeDomain 'AccountK
  116     -- The address key is generated from the account key and address index.
  117     DerivationPath 'AddressK =
  118         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  119 
  120 instance GenMasterKey Byron where
  121     type GenMasterKeyFrom Byron = SomeMnemonic

  123     genMasterKey = generateKeyFromSeed

  125 instance HardDerivation Byron where
  126     type AddressIndexDerivationType Byron = 'WholeDomain
  127     type AccountIndexDerivationType Byron = 'WholeDomain
  128 
  129     deriveAccountPrivateKey pwd rootXPrv idx@(Index accIx) = Byron
  130         { getKey = deriveXPrv DerivationScheme1 pwd (getKey rootXPrv) accIx
  131         , derivationPath = idx
  132         , payloadPassphrase = payloadPassphrase rootXPrv
  133         }
  134 
  135     deriveAddressPrivateKey pwd accXPrv _accStyle idx@(Index addrIx) = Byron
  136         { getKey = deriveXPrv DerivationScheme1 pwd (getKey accXPrv) addrIx
  137         , derivationPath = (derivationPath accXPrv, idx)
  138         , payloadPassphrase = payloadPassphrase accXPrv
  139         }

  142     paymentAddress k = Address
  143         $ CBOR.toStrictByteString

  165 -- | The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  166 minSeedLengthBytes :: Int
  167 minSeedLengthBytes = 16
  168 
  169 -- | Generate a root key from a corresponding seed.
  170 -- The seed should be at least 16 bytes.
  171 generateKeyFromSeed
  172     :: SomeMnemonic
  173     -> ScrubbedBytes
  174     -> Byron 'RootK XPrv
  175 generateKeyFromSeed = unsafeGenerateKeyFromSeed ()
  176 
  177 -- | Generate a new key from seed. Note that the @depth@ is left open so that
  178 -- the caller gets to decide what type of key this is. This is mostly for
  179 -- testing, in practice, seeds are used to represent root keys, and one should
  180 -- use 'generateKeyFromSeed'.
  181 unsafeGenerateKeyFromSeed
  182     :: DerivationPath depth
  183     -> SomeMnemonic
  184     -> ScrubbedBytes
  185     -> Byron depth XPrv
  186 unsafeGenerateKeyFromSeed derivationPath (SomeMnemonic mw) pwd = Byron
  187     { getKey = masterKey
  188     , derivationPath
  189     , payloadPassphrase = hdPassphrase (toXPub masterKey)
  190     }
  191   where
  192     masterKey = generate (hashSeed seedValidated) pwd
  193     seed  = entropyToBytes $ mnemonicToEntropy mw
  194     seedValidated = assert
  195         (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  196         seed
  197 
  198 -- | Hash the seed entropy (generated from mnemonic) used to initiate a HD
  199 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  200 -- minimum for 'generate' (32 bytes).
  201 --
  202 -- Note that our current implementation deviates from BIP-39 because we use a
  203 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  204 --
  205 -- There are two methods of hashing the seed entropy, for different use cases.
  206 --
  207 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  208 --    Blake2b_256, inside a double CBOR serialization sandwich.
  209 --
  210 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  211 --    Blake2b_256, without any serialization.
  212 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  213 hashSeed = serialize . blake2b256 . serialize
  214   where
  215     serialize = BA.convert . cbor . BA.convert
  216     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  217 
  218 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  219 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  220 
  221 -- | Derive a symmetric key for encrypting and authenticating the address
  222 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  223 -- is employed.
  224 hdPassphrase :: XPub -> ScrubbedBytes
  225 hdPassphrase masterKey =
  226     PBKDF2.generate
  227     (PBKDF2.prfHMAC SHA512)
  228     (PBKDF2.Parameters 500 32)
  229     (unXPub masterKey)
  230     ("address-hashing" :: ByteString)
  231 
  232 mkByronKeyFromMasterKey
  233     :: XPrv
  234     -> Byron 'RootK XPrv
  235 mkByronKeyFromMasterKey = unsafeMkByronKeyFromMasterKey ()
  236 
  237 unsafeMkByronKeyFromMasterKey
  238     :: DerivationPath depth
  239     -> XPrv
  240     -> Byron depth XPrv
  241 unsafeMkByronKeyFromMasterKey derivationPath masterKey = Byron
  242     { getKey = masterKey
  243     , derivationPath
  244     , payloadPassphrase = hdPassphrase (toXPub masterKey)
