   91 -- | Material for deriving HD random scheme keys, which can be used for making
   92 -- addresses.
   93 data Byron (depth :: Depth) key = Byron
   94     { getKey :: key
   95     -- ^ The raw private or public key.
   96     , derivationPath :: DerivationPath depth
   97     -- ^ The address derivation indices for the level of this key.
   98     , payloadPassphrase :: ScrubbedBytes
   99     -- ^ Used for encryption of payload containing address derivation path.
  100     } deriving stock (Generic)
  101 
  102 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
  103 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
  104 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
  105 
  106 -- | The hierarchical derivation indices for a given level/depth.
  107 type family DerivationPath (depth :: Depth) :: * where
  108     -- The root key is generated from the seed.
  109     DerivationPath 'RootK =
  110         ()
  111     -- The account key is generated from the root key and account index.
  112     DerivationPath 'AccountK =
  113         Index 'WholeDomain 'AccountK
  114     -- The address key is generated from the account key and address index.
  115     DerivationPath 'AddressK =
  116         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  117 
  118 instance GenMasterKey Byron where
  119     type GenMasterKeyFrom Byron = SomeMnemonic
  120 
  121     genMasterKey = generateKeyFromSeed
  122 
  123 instance HardDerivation Byron where
  124     type AddressIndexDerivationType Byron = 'WholeDomain
  125     type AccountIndexDerivationType Byron = 'WholeDomain
  126 
  127     deriveAccountPrivateKey pwd rootXPrv idx@(Index accIx) = Byron
  128         { getKey = deriveXPrv DerivationScheme1 pwd (getKey rootXPrv) accIx
  129         , derivationPath = idx
  130         , payloadPassphrase = payloadPassphrase rootXPrv
  131         }
  132 
  133     deriveAddressPrivateKey pwd accXPrv _accStyle idx@(Index addrIx) = Byron
  134         { getKey = deriveXPrv DerivationScheme1 pwd (getKey accXPrv) addrIx
  135         , derivationPath = (derivationPath accXPrv, idx)
  136         , payloadPassphrase = payloadPassphrase accXPrv
  137         }

  139 instance PaymentAddress Byron where
  140     paymentAddress discrimination k = unsafeMkAddress
  141         $ CBOR.toStrictByteString
  142         $ CBOR.encodeAddress (getKey k) attrs
  143       where
  144         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  145         pwd = payloadPassphrase k
  146         attrs = case discrimination of
  147             RequiresNoMagic ->
  148                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  149                 ]
  150             RequiresMagic (ProtocolMagic pm) ->
  151                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  152                 , CBOR.encodeProtocolMagicAttr pm
  153                 ]

  159 -- | The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  160 minSeedLengthBytes :: Int
  161 minSeedLengthBytes = 16
  162 
  163 -- | Generate a root key from a corresponding seed.
  164 -- The seed should be at least 16 bytes.
  165 generateKeyFromSeed
  166     :: SomeMnemonic
  167     -> ScrubbedBytes
  168     -> Byron 'RootK XPrv
  169 generateKeyFromSeed = unsafeGenerateKeyFromSeed ()
  170 
  171 -- | Generate a new key from seed. Note that the @depth@ is left open so that
  172 -- the caller gets to decide what type of key this is. This is mostly for
  173 -- testing, in practice, seeds are used to represent root keys, and one should
  174 -- use 'generateKeyFromSeed'.
  175 unsafeGenerateKeyFromSeed
  176     :: DerivationPath depth
  177     -> SomeMnemonic
  178     -> ScrubbedBytes
  179     -> Byron depth XPrv
  180 unsafeGenerateKeyFromSeed derivationPath (SomeMnemonic mw) pwd = Byron
  181     { getKey = masterKey
  182     , derivationPath
  183     , payloadPassphrase = hdPassphrase (toXPub masterKey)
  184     }
  185   where
  186     masterKey = generate (hashSeed seedValidated) pwd
  187     seed  = entropyToBytes $ mnemonicToEntropy mw
  188     seedValidated = assert
  189         (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  190         seed
  191 
  192 -- | Hash the seed entropy (generated from mnemonic) used to initiate a HD
  193 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  194 -- minimum for 'generate' (32 bytes).
  195 --
  196 -- Note that our current implementation deviates from BIP-39 because we use a
  197 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  198 --
  199 -- There are two methods of hashing the seed entropy, for different use cases.
  200 --
  201 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  202 --    Blake2b_256, inside a double CBOR serialization sandwich.
  203 --
  204 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  205 --    Blake2b_256, without any serialization.
  206 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  207 hashSeed = serialize . blake2b256 . serialize
  208   where
  209     serialize = BA.convert . cbor . BA.convert
  210     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  211 
  212 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  213 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  214 
  215 -- | Derive a symmetric key for encrypting and authenticating the address
  216 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  217 -- is employed.
  218 hdPassphrase :: XPub -> ScrubbedBytes
  219 hdPassphrase masterKey =
  220     PBKDF2.generate
  221     (PBKDF2.prfHMAC SHA512)
  222     (PBKDF2.Parameters 500 32)
  223     (unXPub masterKey)
  224     ("address-hashing" :: ByteString)
  225 
  226 mkByronKeyFromMasterKey
  227     :: XPrv
  228     -> Byron 'RootK XPrv
  229 mkByronKeyFromMasterKey = unsafeMkByronKeyFromMasterKey ()
  230 
  231 unsafeMkByronKeyFromMasterKey
  232     :: DerivationPath depth
  233     -> XPrv
  234     -> Byron depth XPrv
  235 unsafeMkByronKeyFromMasterKey derivationPath masterKey = Byron
  236     { getKey = masterKey
  237     , derivationPath
  238     , payloadPassphrase = hdPassphrase (toXPub masterKey)
  239     }
