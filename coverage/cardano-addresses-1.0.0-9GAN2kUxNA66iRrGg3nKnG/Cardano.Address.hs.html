   45 -- | Encoding of addresses for certain key types and backend targets.

   59 -- | Magic constant associated to a given network
   60 newtype ProtocolMagic = ProtocolMagic { getProtocolMagic :: Word32 }
   61     deriving (Generic, Show, Eq)
   62 
   63 -- | Hard-coded protocol magic for the Byron MainNet
   64 mainnetMagic :: ProtocolMagic
   65 mainnetMagic =  ProtocolMagic 764824073
   66 
   67 -- | Derive testnet magic from a type-level Nat
   68 testnetMagic :: forall pm. KnownNat pm => ProtocolMagic
   69 testnetMagic = ProtocolMagic $ fromIntegral $ natVal $ Proxy @pm
