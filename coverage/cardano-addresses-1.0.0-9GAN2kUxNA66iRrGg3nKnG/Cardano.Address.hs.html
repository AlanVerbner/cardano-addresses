   63 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   64 -- irrevelant to the user.
   65 --
   66 -- @since 1.0.0
   67 newtype Address = Address
   68     { unAddress :: ByteString
   69     } deriving stock (Generic, Show, Eq, Ord)
   70       deriving newtype (ByteArrayAccess)
   71 instance NFData Address
   72 
   73 -- Unsafe constructor for easily lifting bytes inside an 'Address'.
   74 --
   75 -- /!\ Use at your own risks.
   76 unsafeMkAddress :: ByteString -> Address
   77 unsafeMkAddress = Address
   78 
   79 -- | Encode an 'Address' to a base58 'Text'.
   80 --
   81 -- @since 1.0.0
   82 base58 :: Address -> Text
   83 base58 = T.decodeUtf8 . encodeBase58 bitcoinAlphabet . unAddress
   84 
   85 -- | Decode a base58-encoded 'Text' into an 'Address'
   86 --
   87 -- @since 1.0.0
   88 fromBase58 :: Text -> Maybe Address
   89 fromBase58 =
   90     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
   91    <=<
   92     decodeBase58 bitcoinAlphabet . T.encodeUtf8
   93 
   94 -- | Encode an 'Address' to bech32 'Text', using @addr@ as a human readable prefix.
   95 --
   96 -- @since 1.0.0
   97 bech32 :: Address -> Text
   98 bech32 = Bech32.encodeLenient hrp . Bech32.dataPartFromBytes . unAddress
   99   where
  100     hrp = [Bech32.humanReadablePart|addr|]
  101 
  102 -- | Decode a bech32-encoded  'Text' into an 'Address'
  103 --
  104 -- @since 1.0.0
  105 fromBech32 :: Text -> Maybe Address
  106 fromBech32 =
  107     fmap unsafeMkAddress . Bech32.dataPartToBytes
  108    <=<
  109     either (const Nothing) (Just . snd) . Bech32.decodeLenient

  121 -- | Describe required network discrimination. See also the available
  122 -- smart-constructors if you are not sure about what to do with this:
  123 --
  124 -- - 'mainnetDiscriminant'
  125 -- - 'stagingDiscriminant'
  126 -- - 'testnetDiscriminant'
  127 --
  128 -- @since 1.0.0
  129 data NetworkDiscriminant
  130     = RequiresNoMagic
  131     | RequiresMagic ProtocolMagic
  132     deriving (Generic, Show, Eq)
  133 instance NFData NetworkDiscriminant

  153 -- | Magic constant associated with a given network. This is mainly used in two
  154 -- places:
  155 --
  156 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  157 -- (2) At the network-level, when doing handshake with nodes.
  158 --
  159 -- @since 1.0.0
  160 newtype ProtocolMagic
  161     = ProtocolMagic { getProtocolMagic :: Word32 }
  162     deriving (Generic, Show, Eq)
  163 instance NFData ProtocolMagic
  164
