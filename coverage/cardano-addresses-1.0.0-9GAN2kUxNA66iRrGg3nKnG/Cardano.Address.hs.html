   52 -- | Address is mere wrapper around a ByteString and represents an encoded address.
   53 newtype Address = Address
   54     { unAddress :: ByteString
   55     } deriving (Generic, Show, Eq, Ord)
   56 instance NFData Address
   57 
   58 -- | Encode an address to base58.
   59 base58 :: Address -> Text
   60 base58 = T.decodeUtf8 . encodeBase58 bitcoinAlphabet . unAddress
   61 
   62 -- | Encode an address to bech32, using @addr@ as a human readable prefix.
   63 bech32 :: Address -> Text
   64 bech32 = Bech32.encodeLenient hrp . Bech32.dataPartFromBytes . unAddress
   65   where
   66     hrp = [Bech32.humanReadablePart|addr|]
   67 
   68 -- | Encoding of addresses for certain key types and backend targets.
   69 class PaymentAddress key where
   70     -- | Convert a public key to a payment 'Address' valid for the given
   71     -- network discrimination.
   72     --
   73     -- Note that 'paymentAddress' is ambiguous and requires therefore a type
   74     -- application.
   75     paymentAddress :: NetworkDiscriminant -> key 'AddressK XPub -> Address

   77 -- | Available network options.
   78 data NetworkDiscriminant
   79     = RequiresNoMagic
   80     | RequiresMagic ProtocolMagic
   81     deriving (Generic, Show, Eq)
   82 instance NFData NetworkDiscriminant
   83 
   84 -- | Required network discrimination settings for mainnet
   85 mainnetDiscriminant :: NetworkDiscriminant
   86 mainnetDiscriminant = RequiresNoMagic
   87 
   88 -- | Required network discrimination settings for staging
   89 stagingDiscriminant :: NetworkDiscriminant
   90 stagingDiscriminant = RequiresNoMagic
   91 
   92 -- | Required network discrimination settings for testnet
   93 testnetDiscriminant :: NetworkDiscriminant
   94 testnetDiscriminant = RequiresMagic testnetMagic
   95 
   96 -- | Magic constant associated to a given network. This is mainly use in two
   97 -- places:
   98 --
   99 -- a) In 'Address' payloads, to discriminate addresses between networks.
  100 -- b) At the network-level, when doing handshake with nodes.
  101 newtype ProtocolMagic
  102     = ProtocolMagic { getProtocolMagic :: Word32 }
  103     deriving (Generic, Show, Eq)
  104 instance NFData ProtocolMagic
  105
