   65 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   66 -- irrevelant to the user.
   67 --
   68 -- @since 1.0.0
   69 newtype Address = Address
   70     { unAddress :: ByteString
   71     } deriving stock (Generic, Show, Eq, Ord)
   72       deriving newtype (ByteArrayAccess)
   73 instance NFData Address
   74 
   75 -- Unsafe constructor for easily lifting bytes inside an 'Address'.
   76 --
   77 -- /!\ Use at your own risks.
   78 unsafeMkAddress :: ByteString -> Address
   79 unsafeMkAddress = Address
   80 
   81 -- | Encode an 'Address' to a base58 'Text'.
   82 --
   83 -- @since 1.0.0
   84 base58 :: Address -> Text
   85 base58 = T.decodeUtf8 . encodeBase58 bitcoinAlphabet . unAddress
   86 
   87 -- | Decode a base58-encoded 'Text' into an 'Address'
   88 --
   89 -- @since 1.0.0
   90 fromBase58 :: Text -> Maybe Address
   91 fromBase58 =
   92     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
   93    <=<
   94     decodeBase58 bitcoinAlphabet . T.encodeUtf8
   95 
   96 -- | Encode an 'Address' to bech32 'Text', using @addr@ as a human readable prefix.
   97 --
   98 -- @since 1.0.0
   99 bech32 :: Address -> Text
  100 bech32 = Bech32.encodeLenient hrp . Bech32.dataPartFromBytes . unAddress
  101   where
  102     hrp = [Bech32.humanReadablePart|addr|]
  103 
  104 -- | Decode a bech32-encoded  'Text' into an 'Address'
  105 --
  106 -- @since 1.0.0
  107 fromBech32 :: Text -> Maybe Address
  108 fromBech32 =
  109     fmap unsafeMkAddress . Bech32.dataPartToBytes
  110    <=<
  111     either (const Nothing) (Just . snd) . Bech32.decodeLenient

  123 -- | Describe required network discrimination. See also the available
  124 -- smart-constructors if you are not sure about what to do with this:
  125 --
  126 -- - 'mainnetDiscriminant'
  127 -- - 'stagingDiscriminant'
  128 -- - 'testnetDiscriminant'
  129 --
  130 -- @since 1.0.0
  131 data NetworkDiscriminant
  132     = RequiresNoMagic
  133     | RequiresMagic ProtocolMagic
  134     deriving (Generic, Show, Eq)
  135 instance NFData NetworkDiscriminant

  155 -- | Magic constant associated with a given network. This is mainly used in two
  156 -- places:
  157 --
  158 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  159 -- (2) At the network-level, when doing handshake with nodes.
  160 --
  161 -- @since 1.0.0
  162 newtype ProtocolMagic
  163     = ProtocolMagic { getProtocolMagic :: Word32 }
  164     deriving (Generic, Show, Eq)
  165 instance NFData ProtocolMagic
  166
