   85 -- | Material for deriving HD random scheme keys, which can be used for making
   86 -- addresses.
   87 data Byron (depth :: Depth) key = Byron
   88     { getKey :: key
   89     -- ^ The raw private or public key.
   90     , derivationPath :: DerivationPath depth
   91     -- ^ The address derivation indices for the level of this key.
   92     , payloadPassphrase :: ScrubbedBytes
   93     -- ^ Used for encryption of payload containing address derivation path.
   94     } deriving stock (Generic)
   95 
   96 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
   97 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
   98 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
   99 
  100 -- | The hierarchical derivation indices for a given level/depth.
  101 type family DerivationPath (depth :: Depth) :: * where
  102     -- The root key is generated from the seed.
  103     DerivationPath 'RootK =
  104         ()
  105     -- The account key is generated from the root key and account index.
  106     DerivationPath 'AccountK =
  107         Index 'WholeDomain 'AccountK
  108     -- The address key is generated from the account key and address index.
  109     DerivationPath 'AddressK =
  110         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  111 
  112 instance GenMasterKey Byron where
  113     type GenMasterKeyFrom Byron = SomeMnemonic
  114 
  115     genMasterKey = generateKeyFromSeed
  116 
  117 instance HardDerivation Byron where
  118     type AddressIndexDerivationType Byron = 'WholeDomain
  119     type AccountIndexDerivationType Byron = 'WholeDomain
  120 
  121     deriveAccountPrivateKey pwd rootXPrv idx@(Index accIx) = Byron
  122         { getKey = deriveXPrv DerivationScheme1 pwd (getKey rootXPrv) accIx
  123         , derivationPath = idx
  124         , payloadPassphrase = payloadPassphrase rootXPrv
  125         }
  126 
  127     deriveAddressPrivateKey pwd accXPrv _accStyle idx@(Index addrIx) = Byron
  128         { getKey = deriveXPrv DerivationScheme1 pwd (getKey accXPrv) addrIx
  129         , derivationPath = (derivationPath accXPrv, idx)
  130         , payloadPassphrase = payloadPassphrase accXPrv
  131         }

  137 -- | The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  138 minSeedLengthBytes :: Int
  139 minSeedLengthBytes = 16
  140 
  141 -- | Generate a root key from a corresponding seed.
  142 -- The seed should be at least 16 bytes.
  143 generateKeyFromSeed
  144     :: SomeMnemonic
  145     -> ScrubbedBytes
  146     -> Byron 'RootK XPrv
  147 generateKeyFromSeed = unsafeGenerateKeyFromSeed ()
  148 
  149 -- | Generate a new key from seed. Note that the @depth@ is left open so that
  150 -- the caller gets to decide what type of key this is. This is mostly for
  151 -- testing, in practice, seeds are used to represent root keys, and one should
  152 -- use 'generateKeyFromSeed'.
  153 unsafeGenerateKeyFromSeed
  154     :: DerivationPath depth
  155     -> SomeMnemonic
  156     -> ScrubbedBytes
  157     -> Byron depth XPrv
  158 unsafeGenerateKeyFromSeed derivationPath (SomeMnemonic mw) pwd = Byron
  159     { getKey = masterKey
  160     , derivationPath
  161     , payloadPassphrase = hdPassphrase (toXPub masterKey)
  162     }
  163   where
  164     masterKey = generate (hashSeed seedValidated) pwd
  165     seed  = entropyToBytes $ mnemonicToEntropy mw
  166     seedValidated = assert
  167         (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  168         seed
  169 
  170 -- | Hash the seed entropy (generated from mnemonic) used to initiate a HD
  171 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  172 -- minimum for 'generate' (32 bytes).
  173 --
  174 -- Note that our current implementation deviates from BIP-39 because we use a
  175 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  176 --
  177 -- There are two methods of hashing the seed entropy, for different use cases.
  178 --
  179 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  180 --    Blake2b_256, inside a double CBOR serialization sandwich.
  181 --
  182 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  183 --    Blake2b_256, without any serialization.
  184 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  185 hashSeed = serialize . blake2b256 . serialize
  186   where
  187     serialize = BA.convert . cbor . BA.convert
  188     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  189 
  190 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  191 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  192 
  193 -- | Derive a symmetric key for encrypting and authenticating the address
  194 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  195 -- is employed.
  196 hdPassphrase :: XPub -> ScrubbedBytes
  197 hdPassphrase masterKey =
  198     PBKDF2.generate
  199     (PBKDF2.prfHMAC SHA512)
  200     (PBKDF2.Parameters 500 32)
  201     (unXPub masterKey)
  202     ("address-hashing" :: ByteString)
  203 
  204 mkByronKeyFromMasterKey
  205     :: XPrv
  206     -> Byron 'RootK XPrv
  207 mkByronKeyFromMasterKey = unsafeMkByronKeyFromMasterKey ()
  208 
  209 unsafeMkByronKeyFromMasterKey
  210     :: DerivationPath depth
  211     -> XPrv
  212     -> Byron depth XPrv
  213 unsafeMkByronKeyFromMasterKey derivationPath masterKey = Byron
  214     { getKey = masterKey
  215     , derivationPath
  216     , payloadPassphrase = hdPassphrase (toXPub masterKey)
  217     }
