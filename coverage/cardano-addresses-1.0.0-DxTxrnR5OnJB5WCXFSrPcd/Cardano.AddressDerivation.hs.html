   75 -- | Key Depth in the derivation path, according to BIP-0039 / BIP-0044
   76 --
   77 -- @m | purpose' | cointype' | account' | change | address@
   78 --
   79 -- We do not manipulate purpose, cointype and change paths directly, so they are
   80 -- left out of the sum type.
   81 data Depth = RootK | AccountK | AddressK
   82 
   83 -- | Marker for addresses type engaged. We want to handle three cases here.
   84 -- The first two are pertinent to UTxO accounting
   85 -- and the last one handles rewards from participation in staking.
   86 -- (a) external chain is used for addresses that are part of the 'advertised'
   87 --     targets of a given transaction
   88 -- (b) internal change is for addresses used to handle the change of a
   89 --     the transaction within a given wallet
   90 -- (c) the addresses for a reward (chimeric) account
   91 data AccountingStyle
   92     = UTxOExternal
   93     | UTxOInternal
   94     | MutableAccount
   95     deriving (Generic, Typeable, Show, Eq, Ord, Bounded)
   96 
   97 instance NFData AccountingStyle
   98 
   99 -- Not deriving 'Enum' because this could have a dramatic impact if we were
  100 -- to assign the wrong index to the corresponding constructor (by swapping
  101 -- around the constructor above for instance).
  102 instance Enum AccountingStyle where
  103     toEnum = \case
  104         0 -> UTxOExternal
  105         1 -> UTxOInternal
  106         2 -> MutableAccount
  107         _ -> error "AccountingStyle.toEnum: bad argument"
  108     fromEnum = \case
  109         UTxOExternal -> 0
  110         UTxOInternal -> 1
  111         MutableAccount -> 2
  112 
  113 -- | A derivation index, with phantom-types to disambiguate derivation type.
  114 --
  115 -- @
  116 -- let accountIx = Index 'Hardened 'AccountK
  117 -- let addressIx = Index 'Soft 'AddressK
  118 -- @
  119 newtype Index (derivationType :: DerivationType) (level :: Depth) = Index
  120     { getIndex :: Word32 }
  121     deriving stock (Generic, Show, Eq, Ord)
  122 
  123 instance NFData (Index derivationType level)
  124 
  125 instance Bounded (Index 'Hardened level) where
  126     minBound = Index 0x80000000
  127     maxBound = Index maxBound
  128 
  129 instance Bounded (Index 'Soft level) where
  130     minBound = Index minBound
  131     maxBound = let (Index ix) = minBound @(Index 'Hardened _) in Index (ix - 1)
  132 
  133 instance Bounded (Index 'WholeDomain level) where
  134     minBound = Index minBound
  135     maxBound = Index maxBound
  136 
  137 instance Enum (Index 'Hardened level) where
  138     fromEnum (Index ix) = fromIntegral ix
  139     toEnum ix
  140         | Index (fromIntegral ix) < minBound @(Index 'Hardened _) =
  141             error "Index@Hardened.toEnum: bad argument"
  142         | otherwise =
  143             Index (fromIntegral ix)
  144 
  145 instance Enum (Index 'Soft level) where
  146     fromEnum (Index ix) = fromIntegral ix
  147     toEnum ix
  148         | Index (fromIntegral ix) > maxBound @(Index 'Soft _) =
  149             error "Index@Soft.toEnum: bad argument"
  150         | otherwise =
  151             Index (fromIntegral ix)
  152 
  153 instance Enum (Index 'WholeDomain level) where
  154     fromEnum (Index ix) = fromIntegral ix
  155     toEnum ix
  156         | Index (fromIntegral ix) > maxBound @(Index 'WholeDomain _) =
  157             error "Index@WholeDomain.toEnum: bad argument"
  158         | otherwise =
  159             Index (fromIntegral ix)
  160 
  161 instance Buildable (Index derivationType level) where
  162     build (Index ix) = fromString (show ix)

  165 -- | Type of derivation that should be used with the given indexes.
  166 --
  167 -- In theory, we should only consider two derivation types: soft and hard.
  168 --
  169 -- However, historically, addresses in Cardano used to be generated across the
  170 -- both soft and hard domain. We therefore introduce a 'WholeDomain' derivation
  171 -- type that is the exact union of `Hardened` and `Soft`.
  172 data DerivationType = Hardened | Soft | WholeDomain
  173 
  174 -- | An interface for doing hard derivations from the root private key, /Master Key/
  175 class HardDerivation (key :: Depth -> * -> *) where
  176     type AccountIndexDerivationType key :: DerivationType
  177     type AddressIndexDerivationType key :: DerivationType
  178 
  179     -- | Derives account private key from the given root private key, using
  180     -- derivation scheme 2 (see <https://github.com/input-output-hk/cardano-crypto/ cardano-crypto>
  181     -- package for more details).
  182     deriveAccountPrivateKey
  183         :: ScrubbedBytes
  184         -> key 'RootK XPrv
  185         -> Index (AccountIndexDerivationType key) 'AccountK
  186         -> key 'AccountK XPrv
  187 
  188     -- | Derives address private key from the given account private key, using
  189     -- derivation scheme 2 (see <https://github.com/input-output-hk/cardano-crypto/ cardano-crypto>
  190     -- package for more details).
  191     deriveAddressPrivateKey
  192         :: ScrubbedBytes
  193         -> key 'AccountK XPrv
  194         -> AccountingStyle
  195         -> Index (AddressIndexDerivationType key) 'AddressK
  196         -> key 'AddressK XPrv
  197 
  198 -- | An interface for doing soft derivations from an account public key
  199 class HardDerivation key => SoftDerivation (key :: Depth -> * -> *) where
  200     -- | Derives address public key from the given account public key, using
  201     -- derivation scheme 2 (see <https://github.com/input-output-hk/cardano-crypto/ cardano-crypto>
  202     -- package for more details).
  203     --
  204     -- This is the preferred way of deriving new sequential address public keys.
  205     deriveAddressPublicKey
  206         :: key 'AccountK XPub
  207         -> AccountingStyle
  208         -> Index 'Soft 'AddressK
  209         -> key 'AddressK XPub

  234 -- | Hard-coded protocol magic for the Byron MainNet
