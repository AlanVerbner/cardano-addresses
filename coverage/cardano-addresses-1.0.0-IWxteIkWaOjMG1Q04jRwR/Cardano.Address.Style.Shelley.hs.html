   80 -- | A cryptographic key for sequential-scheme address derivation, with
   81 -- phantom-types to disambiguate key types.
   82 --
   83 -- @
   84 -- let rootPrivateKey = Shelley 'RootK XPrv
   85 -- let accountPubKey  = Shelley 'AccountK XPub
   86 -- let addressPubKey  = Shelley 'AddressK XPub
   87 -- @
   88 --
   89 -- @since 1.0.0
   90 newtype Shelley (depth :: Depth) key = Shelley
   91     { getKey :: key
   92         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
   93         --
   94         -- @since 1.0.0
   95     }
   96     deriving stock (Generic, Show, Eq)
   97 
   98 deriving instance (Functor (Shelley depth))
   99 instance (NFData key) => NFData (Shelley depth key)
  100 
  101 -- | Unsafe backdoor for constructing an 'Shelley' key from a raw 'XPrv'. this is
  102 -- unsafe because it lets the caller choose the actually derivation 'depth'.
  103 --
  104 -- This can be useful however when serializing / deserializing such a type, or to
  105 -- speed up test code (and avoid having to do needless derivations from a master
  106 -- key down to an address key for instance).
  107 --
  108 -- @since 1.0.0
  109 liftXPrv :: XPrv -> Shelley depth XPrv
  110 liftXPrv = Shelley
  111 
  112 instance GenMasterKey Shelley where
  113     type SecondFactor Shelley = Maybe SomeMnemonic
  114 
  115     genMasterKeyFromXPrv = liftXPrv
  116     genMasterKeyFromMnemonic fstFactor sndFactor =
  117         Shelley $ generateNew seedValidated (maybe mempty mnemonicToBytes sndFactor)
  118         where
  119             mnemonicToBytes (SomeMnemonic mw) = entropyToBytes $ mnemonicToEntropy mw
  120             seed  = mnemonicToBytes fstFactor
  121             seedValidated = assert
  122                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  123                 seed
  124 
  125 instance HardDerivation Shelley where
  126     type AccountIndexDerivationType Shelley = 'Hardened
  127     type AddressIndexDerivationType Shelley = 'Soft
  128     type WithAccountStyle Shelley = AccountingStyle
  129 
  130     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  131         let
  132             purposeIx =
  133                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  134             coinTypeIx =
  135                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  136             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  137                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  138             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  139                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  140             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  141                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  142         in
  143             Shelley acctXPrv
  144 
  145     deriveAddressPrivateKey (Shelley accXPrv) accountingStyle addrIx =
  146         let
  147             changeCode =
  148                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  149             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  150                 deriveXPrv DerivationScheme2 accXPrv changeCode
  151             addrXPrv = -- lvl5 derivation; soft derivation of address index
  152                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  153         in
  154             Shelley addrXPrv
  155 
  156 instance SoftDerivation Shelley where
  157     deriveAddressPublicKey (Shelley accXPub) accountingStyle addrIx =
  158         fromMaybe errWrongIndex $ do
  159             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  160             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  161                 deriveXPub DerivationScheme2 accXPub changeCode
  162             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  163                 deriveXPub DerivationScheme2 changeXPub addrIx
  164             return $ Shelley addrXPub
  165       where
  166         errWrongIndex = error $
  167             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  168             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  169             \either a programmer error, or, we may have reached the maximum \
  170             \number of addresses for a given wallet."

  188 -- | Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  189 -- extension for Cardano:
  190 --
  191 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  192 --
  193 -- It indicates that the subtree of this node is used according to this
  194 -- specification.
  195 --
  196 -- Hardened derivation is used at this level.
  197 purposeIndex :: Word32
  198 purposeIndex = 0x8000073c
  199 
  200 -- | One master node (seed) can be used for unlimited number of independent
  201 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  202 -- same space for various cryptocoins has some disadvantages.
  203 --
  204 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  205 -- addresses across cryptocoins and improving privacy issues.
  206 --
  207 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  208 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  209 -- Lovelace.
  210 --
  211 -- Hardened derivation is used at this level.
  212 coinTypeIndex :: Word32
  213 coinTypeIndex = 0x80000717
  214 
  215 -- | The minimum seed length for 'genMasterKeyFromMnemonic'.
  216 minSeedLengthBytes :: Int
  217 minSeedLengthBytes = 16
