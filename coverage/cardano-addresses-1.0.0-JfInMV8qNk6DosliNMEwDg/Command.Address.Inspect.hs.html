   43 data Cmd = Inspect
   44     deriving (Show)
   45 
   46 mod :: (Cmd -> parent) -> Mod CommandFields parent
   47 mod liftCmd = command "inspect" $
   48     info (helper <*> fmap liftCmd parser) $ mempty
   49         <> progDesc "Show information about an address"
   50         <> footerDoc (Just $ string $ mconcat
   51             [ "The address is read from stdin.\n"
   52             , "\n"
   53             , "Example:\n"
   54             , "  $ cat addr.prv \\\n"
   55             , "  | cardano-address key public \\\n"
   56             , "  | cardano-address address payment --network-tag 0 \\\n"
   57             , "  | cardano-address address delegation $(cat stake.prv | cardano-address key public) \\\n"
   58             , "  | cardano-address address inspect\n"
   59             , "\n"
   60             , "  address style:      Shelley\n"
   61             , "  stake reference:    by value\n"
   62             , "  spending key hash:  79467c69a9ac66280174d09d62575ba955748b21dec3b483a9469a65\n"
   63             , "  stake key hash:     cc339a35f9e0fe039cf510c761d4dd29040c48e9657fdac7e9c01d94\n"
   64             , "  network tag:        0"
   65             ])
   66   where
   67     parser = pure Inspect
   68 
   69 run :: Cmd -> IO ()
   70 run Inspect = do
   71     bytes <- hGetBytes stdin
   72     case inspect (unsafeMkAddress bytes) of
   73         Nothing -> fail "Unrecognized address on standard input"
   74         Just str -> forM_ (T.pack <$> lines str) $ \line -> do
   75             let (label, value) = T.breakOn ":" line
   76             hPutBold     stdout (T.encodeUtf8 label)
   77             B8.hPutStrLn stdout (T.encodeUtf8 value)
   78   where
   79     inspect addr =
   80         inspectByronAddress       addr <|>
   81         inspectIcarusAddress      addr <|>
   82         inspectJormungandrAddress addr <|>
   83         inspectShelleyAddress     addr
