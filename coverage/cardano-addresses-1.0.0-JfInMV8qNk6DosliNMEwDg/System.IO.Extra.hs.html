   77 -- | Read some bytes from the console, and decode them if the encoding is recognized.
   78 hGetBytes :: Handle -> IO ByteString
   79 hGetBytes h = do
   80     raw <- B8.filter (/= '\n') <$> B8.hGetContents h
   81     case detectEncoding (string raw) of
   82         Just (EBase16  ) -> hGetBase16 raw
   83         Just (EBech32{}) -> hGetBech32 raw
   84         Just (EBase58  ) -> hGetBase58 raw
   85         Nothing          -> fail
   86             "Couldn't detect input encoding? Data on stdin must be encoded as \
   87             \bech16, bech32 or base58."
   88   where
   89     detectEncoding :: String -> Maybe (AbstractEncoding ())
   90     detectEncoding str = isBase16 <|> isBech32  <|> isBase58
   91       where
   92         isBase16 = do
   93             guard (all (`elem` "0123456789abcdef") (toLower <$> str))
   94             guard (length str `mod` 2 == 0)
   95             pure EBase16
   96 
   97         isBech32 = do
   98             guard (all (`elem` Bech32.dataCharList) (stripPrefix str))
   99             guard (all (\c -> isLower c || isDigit c) str)
  100             guard ('1' `elem` str)
  101             guard (length (stripPrefix str) > 6)
  102             pure (EBech32 ())
  103           where
  104             stripPrefix = reverse . takeWhile (/= '1') . reverse
  105 
  106         isBase58 = do
  107             guard (all (`elem` string (unAlphabet bitcoinAlphabet)) str)
  108             pure EBase58
  109 
  110     string :: ByteString -> String
  111     string = T.unpack . T.decodeUtf8
  112 
  113     hGetBech32 :: ByteString -> IO ByteString
  114     hGetBech32 raw = either (fail . errToString) pure $ do
  115         (_hrp, dp) <- left Just $ Bech32.decodeLenient $ T.decodeUtf8 raw
  116         maybe (Left Nothing) Right $ Bech32.dataPartToBytes dp
  117       where
  118         unCharPos (Bech32.CharPosition x) = x
  119         invalidCharsMsg = "Invalid character(s) in string"
  120         errToString = ("Bech32 error: " <>) . \case
  121             Just Bech32.StringToDecodeTooLong ->
  122                 "string is too long"
  123             Just Bech32.StringToDecodeTooShort ->
  124                 "string is too short"
  125             Just Bech32.StringToDecodeHasMixedCase ->
  126                 "string has mixed case"
  127             Just Bech32.StringToDecodeMissingSeparatorChar ->
  128                 "string has no separator char"
  129             Just (Bech32.StringToDecodeContainsInvalidChars []) ->
  130                 invalidCharsMsg
  131             Just (Bech32.StringToDecodeContainsInvalidChars ixs) ->
  132                 invalidCharsMsg <> ":\n" <> markCharsRedAtIndices (map unCharPos ixs) (string raw)
  133             Nothing ->
  134                 "invalid data-part; these bytes ain't uint8."
  135 
  136     hGetBase58 :: ByteString -> IO ByteString
  137     hGetBase58 raw = maybe (fail "Invalid Base58-encoded string.") pure $ do
  138         decodeBase58 bitcoinAlphabet raw
  139 
  140     hGetBase16 :: ByteString -> IO ByteString
  141     hGetBase16 raw = either fail pure $ do
  142         convertFromBase Base16 raw

  144 -- | Read some English mnemonic words from the console, or fail.
  145 hGetSomeMnemonic :: Handle -> IO SomeMnemonic
  146 hGetSomeMnemonic h = do
  147     wrds <- T.words . trim '\n' . T.decodeUtf8 <$> B8.hGetContents h
  148     case mkSomeMnemonic @'[ 9, 12, 15, 18, 21, 24 ] wrds of
  149         Left (MkSomeMnemonicError e) -> fail e
  150         Right mw -> pure mw
  151   where
  152     trim c = T.filter (/= c)
  153 
  154 -- | Read an encoded private key from the console, or fail.
  155 hGetXPrv :: Handle -> IO XPrv
  156 hGetXPrv h = do
  157     bytes <- hGetBytes h
  158     case xprvFromBytes bytes of
  159         Nothing  -> fail "Couldn't convert bytes into extended private key."
  160         Just key -> pure key
  161 
  162 -- | Read an encoded public key from the console, or fail.
  163 hGetXPub :: Handle -> IO XPub
  164 hGetXPub h = do
  165     bytes <- hGetBytes h
  166     case xpubFromBytes bytes of
  167         Nothing  -> fail "Couldn't convert bytes into extended public key."
  168         Just key -> pure key
  169 
  170 -- | Read either an encoded public or private key from the console, or fail.
  171 hGetXP__ :: Handle -> IO (Either XPub XPrv)
  172 hGetXP__ h = do
  173     bytes <- hGetBytes h
  174     case (xpubFromBytes bytes, xprvFromBytes bytes) of
  175         (Just xpub,         _) -> pure (Left  xpub)
  176         (_        , Just xprv) -> pure (Right xprv)
  177         _                      -> fail
  178             "Couldn't convert bytes into neither extended public or private keys."

  185 -- | Print bytes to the console with the given encoding.
  186 hPutBytes :: Handle -> ByteString -> Encoding -> IO ()
  187 hPutBytes h bytes = \case
  188     EBase16 ->
  189         B8.hPutStr h $ convertToBase Base16 bytes
  190     EBase58 ->
  191         B8.hPutStr h $ encodeBase58 bitcoinAlphabet bytes
  192     EBech32 hrp ->
  193         B8.hPutStr h $ T.encodeUtf8 $ Bech32.encodeLenient hrp $ Bech32.dataPartFromBytes bytes
  194 
  195 -- | Print bytes to the console, but in bold.
  196 hPutBold :: Handle -> ByteString -> IO ()
  197 hPutBold h bytes =
  198     withSGR h (SetConsoleIntensity BoldIntensity) $ B8.hPutStr h bytes

  205 -- | Fail with a colored red error message.
  206 prettyIOException :: IOException -> IO a
  207 prettyIOException e = do
  208     B8.hPutStrLn stderr $ T.encodeUtf8 $ T.pack $ show e
  209     exitFailure

  211 -- | Bracket-style constructor for applying ANSI Select Graphic Rendition to an
  212 -- action and revert back to normal after.
  213 --
  214 -- This does nothing if the device isn't an ANSI terminal.
  215 withSGR :: Handle -> SGR -> IO a -> IO a
  216 withSGR h sgr action = hIsTerminalDevice h >>= \case
  217     True  -> bracket aFirst aLast aBetween
  218     False -> action
  219   where
  220     aFirst = ([] <$ hSetSGR h [sgr])
  221     aLast = hSetSGR h
  222     aBetween = const action

  224 -- | Mark all characters from a given string as red (in a console).
  225 markCharsRedAtIndices :: Integral i => [i] -> String -> String
  226 markCharsRedAtIndices ixs = go 0 (sort $ nub ixs)
  227   where
  228     go _c [] [] = mempty
  229     go c (i:is) (s:ss)
  230         | c == i    = red ++ s:def ++ go (c + 1) is ss
  231         | otherwise = s : go (c + 1) (i:is) ss
  232     go _ [] ss = ss
  233     go _ _ [] = [] -- NOTE: Really an error case.
  234 
  235     red = setSGRCode [SetColor Foreground Vivid Red]
  236     def = setSGRCode [Reset]
