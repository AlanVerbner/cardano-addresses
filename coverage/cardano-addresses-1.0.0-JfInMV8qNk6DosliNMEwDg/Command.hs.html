   31 import System.Console.ANSI
   32     ( hSupportsANSIWithoutEmulation )
   33 import System.IO
   34     ( BufferMode (..), Handle, hSetBuffering, stderr, stdout )
   35 import System.IO.Extra
   36     ( prettyIOException )
   37 
   38 import qualified Command.Key as Key
   39 import qualified Command.RecoveryPhrase as RecoveryPhrase
   40 
   41 
   42 data CLI
   43     = RecoveryPhrase RecoveryPhrase.Cmd
   44     | Key Key.Cmd
   45     deriving (Show)
   46 
   47 cli :: ParserInfo CLI
   48 cli = info (helper <*> parser) $ progDesc "cardano-addresses"
   49   where
   50     parser = subparser $ mconcat
   51         [ RecoveryPhrase.mod RecoveryPhrase
   52         , Key.mod Key
   53         ]
   54 
   55 -- | Run a given command
   56 run :: CLI -> IO ()
   57 run = handle prettyIOException . \case
   58     RecoveryPhrase sub -> RecoveryPhrase.run sub
   59     Key sub -> Key.run sub
   60 
   61 -- | Parse command line options and arguments
   62 parse :: IO CLI
   63 parse = customExecParser (prefs showHelpOnEmpty) cli
   64 
   65 -- | Enable ANSI colors on Windows and correct output buffering
   66 setup :: IO ()
   67 setup =
   68     mapM_ hSetup [stderr, stdout]
   69   where
   70     hSetup :: Handle -> IO ()
   71     hSetup h = do
   72       void $ hSupportsANSIWithoutEmulation h
   73       hSetBuffering h NoBuffering
