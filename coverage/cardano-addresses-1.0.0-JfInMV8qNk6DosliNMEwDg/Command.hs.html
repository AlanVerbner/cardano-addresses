    6 module Command

    9     -- * I/O interface
   10     , setup

   14 
   15 import Prelude
   16 
   17 import Control.Exception
   18     ( handle )
   19 import Control.Monad
   20     ( void )
   21 import Options.Applicative
   22     ( ParserInfo
   23     , customExecParser
   24     , helper
   25     , info
   26     , prefs
   27     , progDesc
   28     , progDesc
   29     , showHelpOnEmpty
   30     , subparser
   31     )
   32 import System.Console.ANSI
   33     ( hSupportsANSIWithoutEmulation )
   34 import System.IO
   35     ( BufferMode (..), Handle, hSetBuffering, stderr, stdout )
   36 import System.IO.Extra
   37     ( prettyIOException )
   38 
   39 import qualified Command.Key as Key
   40 import qualified Command.RecoveryPhrase as RecoveryPhrase
   41 
   42 
   43 data CLI
   44     = RecoveryPhrase RecoveryPhrase.Cmd
   45     | Key Key.Cmd
   46     deriving (Show)
   47 
   48 cli :: ParserInfo CLI
   49 cli = info (helper <*> parser) $ progDesc "cardano-addresses"
   50   where
   51     parser = subparser $ mconcat
   52         [ RecoveryPhrase.mod RecoveryPhrase
   53         , Key.mod Key
   54         ]
   55 
   56 -- | Run a given command
   57 run :: CLI -> IO ()
   58 run = handle prettyIOException . \case
   59     RecoveryPhrase sub -> RecoveryPhrase.run sub
   60     Key sub -> Key.run sub
   61 
   62 -- | Parse command line options and arguments
   63 parse :: IO CLI
   64 parse = customExecParser (prefs showHelpOnEmpty) cli
   65 
   66 -- | Enable ANSI colors on Windows and correct output buffering
   67 setup :: IO ()
   68 setup =
   69     mapM_ hSetup [stderr, stdout]
   70   where
   71     hSetup :: Handle -> IO ()
   72     hSetup h = do
   73       void $ hSupportsANSIWithoutEmulation h
   74       hSetBuffering h NoBuffering
