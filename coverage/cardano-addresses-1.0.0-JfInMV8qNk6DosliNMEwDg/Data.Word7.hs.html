   37 -- | A 'Word7' algebraic data-type.
   38 -- @since 2.0.0
   39 newtype Word7 = Word7 Word8
   40   deriving (Eq, Show)
   41 
   42 
   43 --
   44 -- Conversions
   45 --
   46 -- > toWord7 1
   47 -- > Word7 1
   48 -- > toWord7 127
   49 -- > Word7 127
   50 -- > toWord7 128
   51 -- > Word7 0
   52 toWord7 :: Word8 -> Word7
   53 toWord7 x = Word7 (x .&. 0x7F)
   54 
   55 toWord8 :: Word7 -> Word8
   56 toWord8 (Word7 x) = x
   57 
   58 -- > toWord7s 1
   59 -- > [Word7 1]
   60 -- > toWord7s 128
   61 -- > [Word7 1,Word7 0]
   62 -- > toWord7s 19099
   63 -- > [Word7 1,Word7 21,Word7 27]
   64 toWord7s :: Natural -> [Word7]
   65 toWord7s = reverse . go
   66     where
   67         go n
   68             | n <= 0x7F = [Word7 . fromIntegral $ n]
   69             | otherwise = (toWord7 . fromIntegral) n : go (shiftR n 7)
   70 
   71 toNatural :: [Word7] -> Natural
   72 toNatural =
   73     fst .
   74     foldr (\(Word7 x) (res, pow) ->
   75                (res + (fromIntegral x)*(limit pow + 1), pow + 7)
   76           )
   77     (0,0)
   78 
   79 limit :: Int -> Natural
   80 limit pow = 2 ^ pow - 1
   81 
   82 --
   83 -- Decoding
   84 --
   85 putVariableLengthNat :: Natural -> Put
   86 putVariableLengthNat = putWord7s . toWord7s
   87 
   88 putWord7s :: [Word7] -> Put
   89 putWord7s [] = pure ()
   90 putWord7s [Word7 x] = putWord8 x
   91 putWord7s (Word7 x : xs) = putWord8 (x .|. 0x80) >> putWord7s xs
