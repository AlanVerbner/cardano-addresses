   58 --
   59 -- Encoding
   60 --
   61 
   62 -- | A concrete 'Encoding' algebraic data-type.
   63 type Encoding = AbstractEncoding HumanReadablePart
   64 
   65 -- | An abstract 'Encoding' to make it easy to map over the bech32 component.
   66 -- Typically used as 'AbstractEncoding HumanReadablePart'.
   67 --
   68 -- > λ> let xpubHRP = [humanReadablePart|xpub|]
   69 -- > λ> let xprvHRP = [humanReadablePart|xprv|]
   70 -- >
   71 -- > λ> fmap (const xpubHRP) (EBech32 xprvHRP)
   72 -- > EBech32 (HumanReadablePart "xpub")
   73 --
   74 data AbstractEncoding a
   75     = EBase16
   76     | EBase58
   77     | EBech32 a
   78     deriving (Eq, Show, Functor)
   79 
   80 --
   81 -- Encode
   82 --
   83 
   84 
   85 -- | Encode a 'ByteString' with the given encoding.
   86 --
   87 -- @since 2.0.0
   88 encode :: Encoding -> ByteString -> ByteString
   89 encode encoding bytes = case encoding of
   90     EBase16 ->
   91         convertToBase Base16 bytes
   92     EBase58 ->
   93         encodeBase58 bitcoinAlphabet bytes
   94     EBech32 hrp ->
   95         T.encodeUtf8 $ Bech32.encodeLenient hrp $ Bech32.dataPartFromBytes bytes

  101 -- | Try detecting the encoding of a given 'String'
  102 --
  103 -- @since 2.0.0
  104 detectEncoding :: String -> Maybe (AbstractEncoding ())
  105 detectEncoding str = isBase16 <|> isBech32  <|> isBase58
  106   where
  107     isBase16 = do
  108         guard (all (`elem` "0123456789abcdef") (toLower <$> str))
  109         guard (length str `mod` 2 == 0)
  110         pure EBase16
  111 
  112     isBech32 = do
  113         guard (all (`elem` Bech32.dataCharList) (stripPrefix str))
  114         guard (all (\c -> isLower c || isDigit c) str)
  115         guard ('1' `elem` str)
  116         guard (length (stripPrefix str) > 6)
  117         pure (EBech32 ())
  118       where
  119         stripPrefix = reverse . takeWhile (/= '1') . reverse
  120 
  121     isBase58 = do
  122         guard (all (`elem` T.unpack (T.decodeUtf8 $ unAlphabet bitcoinAlphabet)) str)
  123         pure EBase58
  124 
  125 -- | Try decoding a base16-encoded 'ByteString'
  126 --
  127 -- @since 2.0.0
  128 fromBase16 :: ByteString -> Either String ByteString
  129 fromBase16 = convertFromBase Base16
  130 
  131 -- | Try decoding a bech32-encoded 'ByteString'
  132 --
  133 -- @since 2.0.0
  134 fromBech32 :: ByteString -> Either String ByteString
  135 fromBech32 raw = left errToString $ do
  136     (_hrp, dp) <- left Just $ Bech32.decodeLenient $ T.decodeUtf8 raw
  137     maybe (Left Nothing) Right $ Bech32.dataPartToBytes dp
  138   where
  139     unCharPos (Bech32.CharPosition x) = x
  140     invalidCharsMsg = "Invalid character(s) in string"
  141     errToString = ("Bech32 error: " <>) . \case
  142         Just Bech32.StringToDecodeTooLong ->
  143             "string is too long"
  144         Just Bech32.StringToDecodeTooShort ->
  145             "string is too short"
  146         Just Bech32.StringToDecodeHasMixedCase ->
  147             "string has mixed case"
  148         Just Bech32.StringToDecodeMissingSeparatorChar ->
  149             "string has no separator char"
  150         Just (Bech32.StringToDecodeContainsInvalidChars []) ->
  151             invalidCharsMsg
  152         Just (Bech32.StringToDecodeContainsInvalidChars ixs) ->
  153             invalidCharsMsg <> ":\n" <> markCharsRedAtIndices
  154                 (map unCharPos ixs)
  155                 (T.unpack . T.decodeUtf8 $ raw)
  156         Nothing ->
  157             "invalid data-part; these bytes ain't uint8."
  158 
  159 fromBase58 :: ByteString -> Either String ByteString
  160 fromBase58 raw = maybe (Left "Invalid Base58-encoded string.") Right $ do
  161     decodeBase58 bitcoinAlphabet raw

  167 -- | Mark all characters from a given string as red (in a console).
  168 markCharsRedAtIndices :: Integral i => [i] -> String -> String
  169 markCharsRedAtIndices ixs = go 0 (sort $ nub ixs)
  170   where
  171     go _c [] [] = mempty
  172     go c (i:is) (s:ss)
  173         | c == i    = red ++ s:def ++ go (c + 1) is ss
  174         | otherwise = s : go (c + 1) (i:is) ss
  175     go _ [] ss = ss
  176     go _ _ [] = [] -- NOTE: Really an error case.
  177 
  178     red = setSGRCode [SetColor Foreground Vivid Red]
  179     def = setSGRCode [Reset]
