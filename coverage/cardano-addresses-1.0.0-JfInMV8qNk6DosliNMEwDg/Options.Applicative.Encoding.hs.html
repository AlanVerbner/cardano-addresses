   29 -- | A concrete 'Encoding' algebraic data-type.
   30 type Encoding = AbstractEncoding HumanReadablePart
   31 
   32 -- | An abstract 'Encoding' to make it easy to map over the bech32 component.
   33 -- Typically used as 'AbstractEncoding HumanReadablePart'.
   34 --
   35 -- > λ> let xpubHRP = [humanReadablePart|xpub|]
   36 -- > λ> let xprvHRP = [humanReadablePart|xprv|]
   37 -- >
   38 -- > λ> fmap (const xpubHRP) (EBech32 xprvHRP)
   39 -- > EBech32 (HumanReadablePart "xpub")
   40 --
   41 data AbstractEncoding a
   42     = EBase16
   43     | EBase58
   44     | EBech32 a
   45     deriving (Eq, Show, Functor)
   46

   48 --
   49 -- Applicative Parser
   50 --
   51 
   52 -- | Parse an 'Encoding' from the command-line, as set of non-overlapping flags.
   53 encodingOpt :: HumanReadablePart -> Parser Encoding
   54 encodingOpt hrp = base16Flag <|> base58Flag <|> bech32Flag
   55   where
   56     base16Flag = flag'  EBase16 (long "base16")
   57     base58Flag = flag'  EBase58 (long "base58")
   58     bech32Flag = flag (EBech32 hrp) (EBech32 hrp) (long "bech32")
