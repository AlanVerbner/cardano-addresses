   71 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   72 -- irrevelant to the user.
   73 --
   74 -- @since 1.0.0
   75 newtype Address = Address
   76     { unAddress :: ByteString
   77     } deriving stock (Generic, Show, Eq, Ord)

   86 -- | Encode an 'Address' to a base58 'Text'.
   87 --
   88 -- @since 1.0.0
   89 base58 :: Address -> Text
   90 base58 = T.decodeUtf8 . encode EBase58 . unAddress
   91 
   92 -- | Decode a base58-encoded 'Text' into an 'Address'
   93 --
   94 -- @since 1.0.0
   95 fromBase58 :: Text -> Maybe Address
   96 fromBase58 =
   97     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
   98    <=<
   99     eitherToMaybe . E.fromBase58 . T.encodeUtf8

  101 -- | Encode an 'Address' to bech32 'Text', using @addr@ as a human readable prefix.
  102 --
  103 -- @since 1.0.0
  104 bech32 :: Address -> Text
  105 bech32 = bech32With hrp
  106   where
  107     hrp = [Bech32.humanReadablePart|addr|]
  108 
  109 -- | Encode an 'Address' to bech32 'Text', using a specified human readable prefix.
  110 --
  111 -- @since 2.0.0
  112 bech32With :: HumanReadablePart -> Address -> Text
  113 bech32With hrp = T.decodeUtf8 . encode (EBech32 hrp) . unAddress
  114 
  115 -- | Decode a bech32-encoded  'Text' into an 'Address'
  116 --
  117 -- @since 1.0.0
  118 fromBech32 :: Text -> Maybe Address
  119 fromBech32 = eitherToMaybe . fmap unsafeMkAddress. E.fromBech32 . T.encodeUtf8

  150 -- | A 'ChainPointer' type representing location of some object
  151 -- in the blockchain (eg., staking certificate). This can be achieved
  152 -- unambiguously by specifying slot number, transaction index and the index
  153 -- in the object list (eg., certification list).
  154 -- For staking certificates, alternatively,  the staking key can be used and
  155 -- then 'DelegationAddress' can be used.
  156 --
  157 -- @since 2.0.0
  158 data ChainPointer = ChainPointer
  159     { slotNum :: Natural
  160       -- ^ Pointer to the slot
  161     , transactionIndex :: Natural
  162       -- ^ transaction index
  163     , outputIndex :: Natural
  164       -- ^ output list index
  165     } deriving stock (Generic, Show, Eq, Ord)
  166 instance NFData ChainPointer

  195 -- Magic constant associated with a given network. This is mainly used in two
  196 -- places:
  197 --
  198 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  199 -- (2) At the network-level, when doing handshake with nodes.
  200 newtype NetworkTag
  201     = NetworkTag Word32
  202     deriving (Generic, Show, Eq)
  203 instance NFData NetworkTag
  204 
  205 -- Describe requirements for address discrimination on the Byron era.
  206 data AddressDiscrimination
  207     = RequiresNetworkTag
  208     | RequiresNoTag
  209     deriving (Generic, Show, Eq)
  210 instance NFData AddressDiscrimination

  212 invariantSize :: HasCallStack => Int -> ByteString -> ByteString
  213 invariantSize expectedLength bytes
  214     | BS.length bytes == expectedLength = bytes
  215     | otherwise = error
  216       $ "length was "
  217       ++ show (BS.length bytes)
  218       ++ ", but expected to be "
  219       ++ (show expectedLength)
  220 
  221 invariantNetworkTag :: HasCallStack => Word32 -> NetworkTag -> Word8
  222 invariantNetworkTag limit (NetworkTag num)
  223     | num < limit = fromIntegral num
  224     | otherwise = error
  225       $ "network tag was "
  226       ++ show num
  227       ++ ", but expected to be less than "
  228       ++ show limit
