   28 import qualified Command.Key.Child as Child

   32 
   33 data Cmd
   34     = FromRecoveryPhrase FromRecoveryPhrase.Cmd
   35     | Child Child.Cmd
   36     | Public Public.Cmd
   37     | Inspect Inspect.Cmd
   38     deriving (Show)
   39 
   40 mod :: (Cmd -> parent) -> Mod CommandFields parent
   41 mod liftCmd = command "key" $
   42     info (helper <*> fmap liftCmd parser) $ mempty
   43         <> progDesc "About public/private keys"
   44         <> footerDoc (Just $ string $ mconcat
   45             [ "Example:\n\n"
   46             , "  $ cardano-address recovery-phrase generate --size 15 \\\n"
   47             , "  | cardano-address key from-recovery-phrase Shelley > root.prv \n"
   48             , "  \n"
   49             , "  $ cat root.prv \\\n"
   50             , "  | cardano-address key child 1852H/1815H/0H \\\n"
   51             , "  | tee acct.prv \\\n"
   52             , "  | cardano-address key public > acct.pub \n"
   53             , "  \n"
   54             , "  $ cat acct.prv | cardano-address key inspect\n"
   55             , "  key type:     private\n"
   56             , "  extended key: 586063c612e8eaa3d807c444c1e5c1630e5a486071edbe18fd900f68236d96438d5f4e3...\n"
   57             , "  chain code:   09b9b219ab1df2bce6597704009cc8ced2fccc9655ff0aebf409b55f0a36ee96\n"
   58             , "  \n"
   59             , "  $ cat acct.pub | cardano-address key inspect\n"
   60             , "  key type:     public\n"
   61             , "  extended key: 406f0caac3a0977684aa4b7182ff62612aa4fbe576c1ba0b9a41c7b11f4a1167\n"
   62             , "  chain code:   09b9b219ab1df2bce6597704009cc8ced2fccc9655ff0aebf409b55f0a36ee96\n"
   63             , "  \n"
   64             , "  $ cat acct.pub | cardano-address key child 0/0 > addr.pub"
   65             ])
   66   where
   67     parser = subparser $ mconcat
   68         [ FromRecoveryPhrase.mod FromRecoveryPhrase
   69         , Child.mod Child
   70         , Public.mod Public
   71         , Inspect.mod Inspect
   72         ]
   73 
   74 run :: Cmd -> IO ()
   75 run = \case
   76     FromRecoveryPhrase sub -> FromRecoveryPhrase.run sub
   77     Child sub -> Child.run sub
   78     Public sub -> Public.run sub
   79     Inspect sub -> Inspect.run sub
