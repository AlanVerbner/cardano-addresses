   27 import qualified Command.Key.Child as Child

   31 
   32 data Cmd
   33     = FromRecoveryPhrase FromRecoveryPhrase.Cmd
   34     | Child Child.Cmd
   35     | Public Public.Cmd
   36     | Inspect Inspect.Cmd
   37     deriving (Show)
   38 
   39 mod :: (Cmd -> parent) -> Mod CommandFields parent
   40 mod liftCmd = command "key" $
   41     info (helper <*> fmap liftCmd parser) $ mempty
   42         <> progDesc "About public/private keys."
   43         <> footerDoc (Just $ string $ mconcat
   44             [ "Example:\n"
   45             , "  $ cardano-address recovery-phrase generate \\\n"
   46             , "  | cardano-address key from-recovery-phrase icarus \\\n"
   47             , "  | cardano-address key public\n"
   48             , "  xpub1k365denpkmqhj9zj6qpax..."
   49             ])
   50   where
   51     parser = subparser $ mconcat
   52         [ FromRecoveryPhrase.mod FromRecoveryPhrase
   53         , Child.mod Child
   54         , Public.mod Public
   55         , Inspect.mod Inspect
   56         ]
   57 
   58 run :: Cmd -> IO ()
   59 run = \case
   60     FromRecoveryPhrase sub -> FromRecoveryPhrase.run sub
   61     Child sub -> Child.run sub
   62     Public sub -> Public.run sub
   63     Inspect sub -> Inspect.run sub
