  100 -- $overview
  101 --
  102 -- This module provides an implementation of:
  103 --
  104 -- - 'GenMasterKey': for generating Shelley master keys from mnemonic sentences
  105 -- - 'HardDerivation': for hierarchical hard derivation of parent to child keys
  106 -- - 'SoftDerivation': for hierarchical soft derivation of parent to child keys
  107 --
  108 -- == Examples
  109 --
  110 -- === Generating a root key from 'SomeMnemonic'
  111 -- > :set -XOverloadedStrings
  112 -- > :set -XTypeApplications
  113 -- > :set -XDataKinds
  114 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  115 -- > import Cardano.Address.Derivation ( GenMasterKey(..) )
  116 -- >
  117 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  118 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  119 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Shelley 'RootK XPrv
  120 --
  121 -- === Generating an 'Address' from a root key
  122 --
  123 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'/0/14@
  124 --
  125 --
  126 -- > import Cardano.Address ( PaymentAddress(..), bech32, mainnetDiscriminant )
  127 -- > import Cardano.Address.Derivation ( AccountingStyle(..), GenMasterKey(..), toXPub )
  128 -- >
  129 -- > let accIx = toEnum 0x80000000
  130 -- > let acctK = deriveAccountPrivateKey rootK accIx
  131 -- >
  132 -- > let addIx = toEnum 0x00000014
  133 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  134 -- >
  135 -- > bech32 $ paymentAddress mainnetDiscriminant (toXPub <$> addrK)
  136 -- > "addr1vzpfffuj3zkp5g7ct6h4va89caxx9ayq2gvkyfvww48sdncxzgg5v"

  141 -- | A cryptographic key for sequential-scheme address derivation, with
  142 -- phantom-types to disambiguate key types.
  143 --
  144 -- @
  145 -- let rootPrivateKey = Shelley 'RootK XPrv
  146 -- let accountPubKey  = Shelley 'AccountK XPub
  147 -- let addressPubKey  = Shelley 'AddressK XPub
  148 -- @
  149 --
  150 -- @since 1.0.0
  151 newtype Shelley (depth :: Depth) key = Shelley
  152     { getKey :: key
  153         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  154         --
  155         -- @since 1.0.0
  156     }
  157     deriving stock (Generic, Show, Eq)
  158 
  159 deriving instance (Functor (Shelley depth))
  160 instance (NFData key) => NFData (Shelley depth key)
  161 
  162 -- | Unsafe backdoor for constructing an 'Shelley' key from a raw 'XPrv'. this is
  163 -- unsafe because it lets the caller choose the actually derivation 'depth'.
  164 --
  165 -- This can be useful however when serializing / deserializing such a type, or to
  166 -- speed up test code (and avoid having to do needless derivations from a master
  167 -- key down to an address key for instance).
  168 --
  169 -- @since 1.0.0
  170 liftXPrv :: XPrv -> Shelley depth XPrv
  171 liftXPrv = Shelley
  172 
  173 instance GenMasterKey Shelley where
  174     type SecondFactor Shelley = ScrubbedBytes
  175 
  176     genMasterKeyFromXPrv = liftXPrv
  177     genMasterKeyFromMnemonic fstFactor sndFactor =
  178         Shelley $ generateNew seedValidated sndFactor
  179         where
  180             seed  = someMnemonicToBytes fstFactor
  181             seedValidated = assert
  182                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  183                 seed

  185 instance HardDerivation Shelley where
  186     type AccountIndexDerivationType Shelley = 'Hardened
  187     type AddressIndexDerivationType Shelley = 'Soft
  188     type WithAccountStyle Shelley = AccountingStyle
  189 
  190     deriveAccountPrivateKey (Shelley rootXPrv) accIx =
  191         let
  192             purposeIx =
  193                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  194             coinTypeIx =
  195                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  196             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  197                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  198             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  199                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  200             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  201                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  202         in
  203             Shelley acctXPrv
  204 
  205     deriveAddressPrivateKey (Shelley accXPrv) accountingStyle addrIx =
  206         let
  207             changeCode =
  208                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  209             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  210                 deriveXPrv DerivationScheme2 accXPrv changeCode
  211             addrXPrv = -- lvl5 derivation; soft derivation of address index
  212                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  213         in
  214             Shelley addrXPrv
  215 
  216 -- | Derive a staking key for a corresponding 'AccountK'. Note that wallet
  217 -- software are by convention only using one staking key per account, and always
  218 -- the first account (with index 0').
  219 --
  220 -- Deriving staking keys for something else than the initial account is not
  221 -- recommended and can lead to incompatibility with existing wallet softwares
  222 -- (Daedalus, Yoroi, Adalite...).
  223 deriveStakingPrivateKey
  224     :: Shelley 'AccountK XPrv
  225     -> Shelley 'StakingK XPrv
  226 deriveStakingPrivateKey (Shelley accXPrv) =
  227     let
  228         changeXPrv = -- lvl4 derivation; soft derivation of change chain
  229             deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  230         stakeXPrv = -- lvl5 derivation; soft derivation of address index
  231             deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  232     in
  233         Shelley stakeXPrv
  234 
  235 instance SoftDerivation Shelley where
  236     deriveAddressPublicKey (Shelley accXPub) accountingStyle addrIx =
  237         fromMaybe errWrongIndex $ do
  238             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  239             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  240                 deriveXPub DerivationScheme2 accXPub changeCode
  241             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  242                 deriveXPub DerivationScheme2 changeXPub addrIx
  243             return $ Shelley addrXPub
  244       where
  245         errWrongIndex = error $
  246             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  247             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  248             \either a programmer error, or, we may have reached the maximum \
  249             \number of addresses for a given wallet."

  256 -- | Error reported from trying to create a network discriminant from number
  257 --
  258 -- @since 1.0.0
  259 newtype MkNetworkDiscriminantError
  260     = ErrWrongNetworkTag Word8
  261       -- ^ Wrong network tag.
  262     deriving (Eq, Show)

  264 mkNetworkDiscriminant
  265     :: Word8
  266     -> Either MkNetworkDiscriminantError (NetworkDiscriminant Shelley)
  267 mkNetworkDiscriminant nTag
  268     | nTag < 16 =  Right $ NetworkTag $ fromIntegral nTag
  269     | otherwise = Left $ ErrWrongNetworkTag nTag

  271 instance PaymentAddress Shelley where
  272     paymentAddress discrimination k = unsafeMkAddress $
  273         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  274             putWord8 firstByte
  275             putByteString (blake2b224 k)
  276       where
  277           -- we use here the fact that payment address stands for what is named
  278           -- as enterprise address, ie., address carrying no stake rights. For
  279           -- rationale why we may need such addresses refer to delegation
  280           -- specification - Section 3.2.3. What is important here is that the
  281           -- address is composed of discrimination byte and 28 bytes hashed public key.
  282           -- Moreover, it was decided that first 4 bits for enterprise address
  283           -- will be `0110`. The next for 4 bits are reserved for network discriminator.
  284           -- `0110 0000` is 96 in decimal.
  285           firstByte =
  286               96 + invariantNetworkTag (networkTag @Shelley discrimination)
  287           expectedLength = 1 + publicKeyHashSize
  288 
  289 instance DelegationAddress Shelley where
  290     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  291         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  292             putWord8 firstByte
  293             putByteString . blake2b224 $ paymentKey
  294             putByteString . blake2b224 $ stakingKey
  295       where
  296           -- we use here the fact that payment address stands for what is named
  297           -- as base address - refer to delegation specification - Section 3.2.1.
  298           -- What is important here is that the address is composed of discrimination
  299           -- byte and two 28 bytes hashed public keys, one for payment key and the
  300           -- other for staking/reword key.
  301           -- Moreover, it was decided that first 4 bits for enterprise address
  302           -- will be `0000`. The next for bits are reserved for network discriminator.
  303           firstByte =
  304               invariantNetworkTag (networkTag @Shelley discrimination)
  305           expectedLength = 1 + 2*publicKeyHashSize
  306 
  307 invariantNetworkTag :: HasCallStack => NetworkTag -> Word8
  308 invariantNetworkTag (NetworkTag num)
  309     | num < 16 = fromIntegral num
  310     | otherwise = error
  311       $ "network tag was "
  312       ++ show num
  313       ++ ", but expected to be less than 16"
  314 
  315 invariantSize :: HasCallStack => Int -> ByteString -> ByteString
  316 invariantSize expectedLength bytes
  317     | BS.length bytes == expectedLength = bytes
  318     | otherwise = error
  319       $ "length was "
  320       ++ show (BS.length bytes)
  321       ++ ", but expected to be "
  322       ++ (show expectedLength)
  323 
  324 blake2b224 :: Shelley depth XPub -> ByteString
  325 blake2b224 =
  326     BA.convert . hash @_ @Blake2b_224 . getPublicKey . getKey

  332 -- | Size, in bytes, of a hash of public key (without the corresponding chain code)
  333 publicKeyHashSize :: Int
  334 publicKeyHashSize = hashDigestSize Blake2b_224

  336 -- | Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  337 -- extension for Cardano:
  338 --
  339 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  340 --
  341 -- It indicates that the subtree of this node is used according to this
  342 -- specification.
  343 --
  344 -- Hardened derivation is used at this level.
  345 purposeIndex :: Word32
  346 purposeIndex = 0x8000073c
  347 
  348 -- | One master node (seed) can be used for unlimited number of independent
  349 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  350 -- same space for various cryptocoins has some disadvantages.
  351 --
  352 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  353 -- addresses across cryptocoins and improving privacy issues.
  354 --
  355 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  356 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  357 -- Lovelace.
  358 --
  359 -- Hardened derivation is used at this level.
  360 coinTypeIndex :: Word32
  361 coinTypeIndex = 0x80000717
