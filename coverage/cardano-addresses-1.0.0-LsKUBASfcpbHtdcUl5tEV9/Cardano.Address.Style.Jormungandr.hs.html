   89 -- $overview
   90 --
   91 -- This module provides an implementation of:
   92 --
   93 -- - 'GenMasterKey': for generating Jormungandr master keys from mnemonic sentences
   94 -- - 'HardDerivation': for hierarchical hard derivation of parent to child keys
   95 -- - 'SoftDerivation': for hierarchical soft derivation of parent to child keys
   96 -- - 'PaymentAddress': for constructing payment addresses from a address public key
   97 -- - 'DelegationAddress': for constructing payment addresses from a address and stake public keys
   98 --
   99 -- == Examples
  100 --
  101 -- === Generating a root key from 'SomeMnemonic'
  102 -- > :set -XOverloadedStrings
  103 -- > :set -XTypeApplications
  104 -- > :set -XDataKinds
  105 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  106 -- > import Cardano.Address.Derivation ( GenMasterKey(..) )
  107 -- >
  108 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  109 -- > let sndFactor = mempty -- Or alternatively, a second factor mnemonic transformed to bytes via someMnemonicToBytes
  110 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Jormungandr 'RootK XPrv
  111 --
  112 -- === Generating an 'Address' from a root key
  113 --
  114 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'/0/14@
  115 --
  116 --
  117 -- > import Cardano.Address ( PaymentAddress(..), DelegationAddress(..), bech32 )
  118 -- > import Cardano.Address.Derivation ( AccountingStyle(..), StakingDerivation (..), GenMasterKey(..), toXPub )
  119 -- >
  120 -- > let accIx = toEnum 0x80000000
  121 -- > let acctK = deriveAccountPrivateKey rootK accIx
  122 -- >
  123 -- > let addIx = toEnum 0x00000014
  124 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  125 -- >
  126 -- > bech32 $ paymentAddress jormungandrTestnet (toXPub <$> addrK)
  127 -- > "addr1s0lgjsr0kjsprvkxmetgcjaxsq833rxg3g8rv528wa0l5c8wcnplq3x0w2h"
  128 -- >
  129 -- > let stakeK = deriveStakingPrivateKey acctK
  130 -- > bech32 $ delegationAddress jormungandrTestnet (toXPub <$> addrK) (toXPub <$> stakeK)
  131 -- > "addr1snlgjsr0kjsprvkxmetgcjaxsq833rxg3g8rv528wa0l5c8wcnplq9t56crgcdw8wmat0tqwj3zqeqekgs0v9hjuy85lr6pgfmy3hxh0nedreq"

  136 -- | A cryptographic key for sequential-scheme address derivation, with
  137 -- phantom-types to disambiguate key types.
  138 --
  139 -- @
  140 -- let rootPrivateKey = Jormungandr 'RootK XPrv
  141 -- let accountPubKey  = Jormungandr 'AccountK XPub
  142 -- let addressPubKey  = Jormungandr 'AddressK XPub
  143 -- @
  144 --
  145 -- @since 2.0.0
  146 newtype Jormungandr (depth :: Depth) key = Jormungandr
  147     { getKey :: key
  148         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  149         --
  150         -- @since 2.0.0
  151     }
  152     deriving stock (Generic, Show, Eq)
  153 
  154 deriving instance (Functor (Jormungandr depth))
  155 instance (NFData key) => NFData (Jormungandr depth key)
  156 
  157 -- | Unsafe backdoor for constructing an 'Jormungandr' key from a raw 'XPrv'. this is
  158 -- unsafe because it lets the caller choose the actually derivation 'depth'.
  159 --
  160 -- This can be useful however when serializing / deserializing such a type, or to
  161 -- speed up test code (and avoid having to do needless derivations from a master
  162 -- key down to an address key for instance).
  163 --
  164 -- @since 2.0.0
  165 liftXPrv :: XPrv -> Jormungandr depth XPrv
  166 liftXPrv = Jormungandr

  168 instance GenMasterKey Jormungandr where
  169     type SecondFactor Jormungandr = ScrubbedBytes
  170 
  171     genMasterKeyFromXPrv = liftXPrv
  172     genMasterKeyFromMnemonic fstFactor sndFactor =
  173         Jormungandr $ generateNew seedValidated sndFactor
  174         where
  175             seed  = someMnemonicToBytes fstFactor
  176             seedValidated = assert
  177                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  178                 seed

  180 instance HardDerivation Jormungandr where
  181     type AccountIndexDerivationType Jormungandr = 'Hardened
  182     type AddressIndexDerivationType Jormungandr = 'Soft
  183     type WithAccountStyle Jormungandr = AccountingStyle
  184 
  185     deriveAccountPrivateKey (Jormungandr rootXPrv) accIx =
  186         let
  187             purposeIx =
  188                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  189             coinTypeIx =
  190                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  191             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  192                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  193             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  194                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  195             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  196                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  197         in
  198             Jormungandr acctXPrv
  199 
  200     deriveAddressPrivateKey (Jormungandr accXPrv) accountingStyle addrIx =
  201         let
  202             changeCode =
  203                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  204             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  205                 deriveXPrv DerivationScheme2 accXPrv changeCode
  206             addrXPrv = -- lvl5 derivation; soft derivation of address index
  207                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  208         in
  209             Jormungandr addrXPrv
  210 
  211 instance SoftDerivation Jormungandr where
  212     deriveAddressPublicKey (Jormungandr accXPub) accountingStyle addrIx =
  213         fromMaybe errWrongIndex $ do
  214             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  215             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  216                 deriveXPub DerivationScheme2 accXPub changeCode
  217             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  218                 deriveXPub DerivationScheme2 changeXPub addrIx
  219             return $ Jormungandr addrXPub
  220       where
  221         errWrongIndex = error $
  222             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  223             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  224             \either a programmer error, or, we may have reached the maximum \
  225             \number of addresses for a given wallet."
  226 
  227 instance StakingDerivation Jormungandr where
  228     deriveStakingPrivateKey (Jormungandr accXPrv) =
  229         let
  230             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  231                 deriveXPrv DerivationScheme2 accXPrv (toEnum @(Index 'Soft _) 2)
  232             stakeXPrv = -- lvl5 derivation; soft derivation of address index
  233                 deriveXPrv DerivationScheme2 changeXPrv (minBound @(Index 'Soft _))
  234         in
  235             Jormungandr stakeXPrv

  240 incentivizedTestnet :: NetworkDiscriminant Jormungandr
  241 incentivizedTestnet = NetworkTag 0x83

  243 instance HasNetworkDiscriminant Jormungandr where
  244     type NetworkDiscriminant Jormungandr = NetworkTag
  245     addressDiscrimination _ = RequiresNetworkTag
  246     networkTag = id
  247 
  248 instance PaymentAddress Jormungandr where
  249     paymentAddress discrimination k = unsafeMkAddress $
  250         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  251             putWord8 (invariantNetworkTag 255 firstByte)
  252             putByteString (getPublicKey $ getKey k)
  253       where
  254           firstByte = networkTag @Jormungandr discrimination
  255           expectedLength = 1 + publicKeySize
  256 
  257 instance DelegationAddress Jormungandr where
  258     delegationAddress discrimination paymentKey stakingKey = unsafeMkAddress $
  259         invariantSize expectedLength $ BL.toStrict $ runPut $ do
  260             putWord8 (invariantNetworkTag 255 $ NetworkTag $ firstByte + 1)
  261             putByteString . getPublicKey . getKey $ paymentKey
  262             putByteString . getPublicKey . getKey $ stakingKey
  263       where
  264           (NetworkTag firstByte) = networkTag @Jormungandr discrimination
  265           expectedLength = 1 + 2*publicKeySize

  283 -- Purpose is a constant set to 1852' (or 0x8000073c) following the BIP-44
  284 -- extension for Cardano:
  285 --
  286 -- https://github.com/input-output-hk/implementation-decisions/blob/e2d1bed5e617f0907bc5e12cf1c3f3302a4a7c42/text/1852-hd-chimeric.md
  287 --
  288 -- It indicates that the subtree of this node is used according to this
  289 -- specification.
  290 --
  291 -- Hardened derivation is used at this level.
  292 purposeIndex :: Word32
  293 purposeIndex = 0x8000073c
  294 
  295 -- One master node (seed) can be used for unlimited number of independent
  296 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  297 -- same space for various cryptocoins has some disadvantages.
  298 --
  299 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  300 -- addresses across cryptocoins and improving privacy issues.
  301 --
  302 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  303 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  304 -- Lovelace.
  305 --
  306 -- Hardened derivation is used at this level.
  307 coinTypeIndex :: Word32
  308 coinTypeIndex = 0x80000717
