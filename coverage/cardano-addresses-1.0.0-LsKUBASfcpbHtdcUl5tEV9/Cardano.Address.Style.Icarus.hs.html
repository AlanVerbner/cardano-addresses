   99 -- $overview
  100 --
  101 -- This module provides an implementation of:
  102 --
  103 -- - 'GenMasterKey': for generating Icarus master keys from mnemonic sentences
  104 -- - 'HardDerivation': for hierarchical hard derivation of parent to child keys
  105 -- - 'SoftDerivation': for hierarchical soft derivation of parent to child keys
  106 -- - 'PaymentAddress': for constructing addresses from a public key
  107 --
  108 -- We call 'Icarus' addresses the new format of Cardano addresses which came
  109 -- after 'Byron'. This is the format used by /Yoroi/ and now also used by
  110 -- /Daedalus/.
  111 --
  112 -- == Examples
  113 --
  114 -- === Generating a root key from 'SomeMnemonic'
  115 -- > :set -XOverloadedStrings
  116 -- > :set -XTypeApplications
  117 -- > :set -XDataKinds
  118 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  119 -- > import Cardano.Address.Derivation ( GenMasterKey(..) )
  120 -- >
  121 -- > let (Right mw) = mkSomeMnemonic @'[15] ["network","empty","cause","mean","expire","private","finger","accident","session","problem","absurd","banner","stage","void","what"]
  122 -- > let sndFactor = mempty -- Or alternatively, a second factor passphrase
  123 -- > let rootK = genMasterKeyFromMnemonic mw sndFactor :: Icarus 'RootK XPrv
  124 --
  125 -- === Generating an 'Address' from a root key
  126 --
  127 -- Let's consider the following 3rd, 4th and 5th derivation paths @0'/0/14@
  128 --
  129 --
  130 -- > import Cardano.Address ( PaymentAddress(..), base58, mainnetDiscriminant )
  131 -- > import Cardano.Address.Derivation ( AccountingStyle(..), GenMasterKey(..), toXPub )
  132 -- >
  133 -- > let accIx = toEnum 0x80000000
  134 -- > let acctK = deriveAccountPrivateKey rootK accIx
  135 -- >
  136 -- > let addIx = toEnum 0x00000014
  137 -- > let addrK = deriveAddressPrivateKey acctK UTxOExternal addIx
  138 -- >
  139 -- > base58 $ paymentAddress mainnetDiscriminant (toXPub <$> addrK)
  140 -- > "Ae2tdPwUPEZ8XpsjgQPH2cJdtohkYrxJ3i5y6mVsrkZZkdpdn6mnr4Rt6wG"

  145 -- | A cryptographic key for sequential-scheme address derivation, with
  146 -- phantom-types to disambiguate key types.
  147 --
  148 -- @
  149 -- let rootPrivateKey = Icarus 'RootK XPrv
  150 -- let accountPubKey  = Icarus 'AccountK XPub
  151 -- let addressPubKey  = Icarus 'AddressK XPub
  152 -- @
  153 --
  154 -- @since 1.0.0
  155 newtype Icarus (depth :: Depth) key = Icarus
  156     { getKey :: key
  157         -- ^ Extract the raw 'XPrv' or 'XPub' wrapped by this type.
  158         --
  159         -- @since 1.0.0
  160     }
  161     deriving stock (Generic, Show, Eq)
  162 
  163 deriving instance (Functor (Icarus depth))
  164 instance (NFData key) => NFData (Icarus depth key)

  192 -- | Unsafe backdoor for constructing an 'Icarus' key from a raw 'XPrv'. this is
  193 -- unsafe because it lets the caller choose the actually derivation 'depth'.
  194 --
  195 -- This can be useful however when serializing / deserializing such a type, or to
  196 -- speed up test code (and avoid having to do needless derivations from a master
  197 -- key down to an address key for instance).
  198 --
  199 -- @since 1.0.0
  200 liftXPrv :: XPrv -> Icarus depth XPrv
  201 liftXPrv = Icarus

  203 instance GenMasterKey Icarus where

  206     genMasterKeyFromXPrv = liftXPrv
  207     genMasterKeyFromMnemonic (SomeMnemonic mw) sndFactor =
  208         let
  209             seed  = entropyToBytes $ mnemonicToEntropy mw
  210             seedValidated = assert
  211                 (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  212                 seed
  213         in Icarus $ generateNew seedValidated sndFactor

  215 instance HardDerivation Icarus where
  216     type AccountIndexDerivationType Icarus = 'Hardened
  217     type AddressIndexDerivationType Icarus = 'Soft
  218     type WithAccountStyle Icarus = AccountingStyle
  219 
  220     deriveAccountPrivateKey (Icarus rootXPrv) accIx =
  221         let
  222             purposeIx =
  223                 toEnum @(Index 'Hardened _) $ fromEnum purposeIndex
  224             coinTypeIx =
  225                 toEnum @(Index 'Hardened _) $ fromEnum coinTypeIndex
  226             purposeXPrv = -- lvl1 derivation; hardened derivation of purpose'
  227                 deriveXPrv DerivationScheme2 rootXPrv purposeIx
  228             coinTypeXPrv = -- lvl2 derivation; hardened derivation of coin_type'
  229                 deriveXPrv DerivationScheme2 purposeXPrv coinTypeIx
  230             acctXPrv = -- lvl3 derivation; hardened derivation of account' index
  231                 deriveXPrv DerivationScheme2 coinTypeXPrv accIx
  232         in
  233             Icarus acctXPrv
  234 
  235     deriveAddressPrivateKey (Icarus accXPrv) accountingStyle addrIx =
  236         let
  237             changeCode =
  238                 toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  239             changeXPrv = -- lvl4 derivation; soft derivation of change chain
  240                 deriveXPrv DerivationScheme2 accXPrv changeCode
  241             addrXPrv = -- lvl5 derivation; soft derivation of address index
  242                 deriveXPrv DerivationScheme2 changeXPrv addrIx
  243         in
  244             Icarus addrXPrv
  245 
  246 instance SoftDerivation Icarus where
  247     deriveAddressPublicKey (Icarus accXPub) accountingStyle addrIx =
  248         fromMaybe errWrongIndex $ do
  249             let changeCode = toEnum @(Index 'Soft _) $ fromEnum accountingStyle
  250             changeXPub <- -- lvl4 derivation in bip44 is derivation of change chain
  251                 deriveXPub DerivationScheme2 accXPub changeCode
  252             addrXPub <- -- lvl5 derivation in bip44 is derivation of address chain
  253                 deriveXPub DerivationScheme2 changeXPub addrIx
  254             return $ Icarus addrXPub
  255       where
  256         errWrongIndex = error $
  257             "deriveAddressPublicKey failed: was given an hardened (or too big) \
  258             \index for soft path derivation ( " ++ show addrIx ++ "). This is \
  259             \either a programmer error, or, we may have reached the maximum \
  260             \number of addresses for a given wallet."

  286 -- | Purpose is a constant set to 44' (or 0x8000002C) following the original
  287 -- BIP-44 specification.
  288 --
  289 -- It indicates that the subtree of this node is used according to this
  290 -- specification.
  291 --
  292 -- Hardened derivation is used at this level.
  293 purposeIndex :: Word32
  294 purposeIndex = 0x8000002C
  295 
  296 -- | One master node (seed) can be used for unlimited number of independent
  297 -- cryptocoins such as Bitcoin, Litecoin or Namecoin. However, sharing the
  298 -- same space for various cryptocoins has some disadvantages.
  299 --
  300 -- This level creates a separate subtree for every cryptocoin, avoiding reusing
  301 -- addresses across cryptocoins and improving privacy issues.
  302 --
  303 -- Coin type is a constant, set for each cryptocoin. For Cardano this constant
  304 -- is set to 1815' (or 0x80000717). 1815 is the birthyear of our beloved Ada
  305 -- Lovelace.
  306 --
  307 -- Hardened derivation is used at this level.
  308 coinTypeIndex :: Word32
  309 coinTypeIndex = 0x80000717
  310 
  311 -- The minimum seed length for 'generateKeyFromMnemonic' and 'unsafeGenerateKeyFromMnemonic'.
  312 minSeedLengthBytes :: Int
  313 minSeedLengthBytes = 16
  314 
  315 -- Hardware Ledger devices generates keys from mnemonic using a different
  316 -- approach (different from the rest of Cardano).
  317 --
  318 -- It is a combination of:
  319 --
  320 -- - [SLIP 0010](https://github.com/satoshilabs/slips/blob/master/slip-0010.md)
  321 -- - [BIP 0032](https://github.com/bitcoin/bips/blob/master/bip-0032.mediawiki)
  322 -- - [BIP 0039](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki)
  323 -- - [RFC 8032](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  324 -- - What seems to be arbitrary changes from Ledger regarding the calculation of
  325 --   the initial chain code and generation of the root private key.
  326 unsafeGenerateKeyFromHardwareLedger
  327     :: SomeMnemonic
  328         -- ^ The root mnemonic
  329     -> Icarus 'RootK XPrv
  330 unsafeGenerateKeyFromHardwareLedger (SomeMnemonic mw) = unsafeFromRight $ do
  331     let seed = pbkdf2HmacSha512
  332             $ T.encodeUtf8
  333             $ T.intercalate " "
  334             $ mnemonicToText mw
  335 
  336     -- NOTE
  337     -- SLIP-0010 refers to `iR` as the chain code. Here however, the chain code
  338     -- is obtained as a hash of the initial seed whereas iR is used to make part
  339     -- of the root private key itself.
  340     let cc = hmacSha256 (BS.pack [1] <> seed)
  341     let (iL, iR) = first pruneBuffer $ hashRepeatedly seed
  342 
  343     prv <- maybe (Left "invalid xprv") pure $ xprvFromBytes $ iL <> iR <> cc
  344     pure $ Icarus prv
  345   where
  346     -- Errors yielded in the body of 'unsafeGenerateKeyFromHardwareLedger' are
  347     -- programmer errors (out-of-range byte buffer access or, invalid length for
  348     -- cryptographic operations). Therefore, we throw badly if we encounter any.
  349     unsafeFromRight :: Either String a -> a
  350     unsafeFromRight = either error id
  351 
  352     -- This is the algorithm described in SLIP 0010 for master key generation
  353     -- with an extra step to discard _some_ of the potential private keys. Why
  354     -- this extra step remains a mystery as of today.
  355     --
  356     --      1. Generate a seed byte sequence S of 512 bits according to BIP-0039.
  357     --         (done in a previous step, passed as argument).
  358     --
  359     --      2. Calculate I = HMAC-SHA512(Key = "ed25519 seed", Data = S)
  360     --
  361     --      3. Split I into two 32-byte sequences, IL and IR.
  362     --
  363     -- extra *******************************************************************
  364     -- *                                                                       *
  365     -- *    3.5 If the third highest bit of the last byte of IL is not zero    *
  366     -- *        S = I and go back to step 2.                                   *
  367     -- *                                                                       *
  368     -- *************************************************************************
  369     --
  370     --      4. Use parse256(IL) as master secret key, and IR as master chain code.
  371     hashRepeatedly :: ByteString -> (ByteString, ByteString)
  372     hashRepeatedly bytes = case BS.splitAt 32 (hmacSha512 bytes) of
  373         (iL, iR) | isInvalidKey iL -> hashRepeatedly (iL <> iR)
  374         (iL, iR) -> (iL, iR)
  375       where
  376         isInvalidKey k = testBit (k `BS.index` 31) 5
  377 
  378     -- - Clear the lowest 3 bits of the first byte
  379     -- - Clear the highest bit of the last byte
  380     -- - Set the second highest bit of the last byte
  381     --
  382     -- As described in [RFC 8032 - 5.1.5](https://tools.ietf.org/html/rfc8032#section-5.1.5)
  383     pruneBuffer :: ByteString -> ByteString
  384     pruneBuffer bytes =
  385         let
  386             (firstByte, rest) = fromMaybe (error "pruneBuffer: no first byte") $
  387                 BS.uncons bytes
  388 
  389             (rest', lastByte) = fromMaybe (error "pruneBuffer: no last byte") $
  390                 BS.unsnoc rest
  391 
  392             firstPruned = firstByte
  393                 & (`clearBit` 0)
  394                 & (`clearBit` 1)
  395                 & (`clearBit` 2)
  396 
  397             lastPruned = lastByte
  398                 & (`setBit` 6)
  399                 & (`clearBit` 7)
  400         in
  401             (firstPruned `BS.cons` BS.snoc rest' lastPruned)
  402 
  403     -- As described in [BIP 0039 - From Mnemonic to Seed](https://github.com/bitcoin/bips/blob/master/bip-0039.mediawiki#from-mnemonic-to-seed)
  404     pbkdf2HmacSha512 :: ByteString -> ByteString
  405     pbkdf2HmacSha512 bytes = PBKDF2.generate
  406         (PBKDF2.prfHMAC SHA512)
  407         (PBKDF2.Parameters 2048 64)
  408         bytes
  409         ("mnemonic" :: ByteString)
  410 
  411     hmacSha256 :: ByteString -> ByteString
  412     hmacSha256 =
  413         BA.convert @(HMAC SHA256) . hmac salt
  414 
  415     -- As described in [SLIP 0010 - Master Key Generation](https://github.com/satoshilabs/slips/blob/master/slip-0010.md#master-key-generation)
  416     hmacSha512 :: ByteString -> ByteString
  417     hmacSha512 =
  418         BA.convert @(HMAC SHA512) . hmac salt
  419 
  420     salt :: ByteString
  421     salt = "ed25519 seed"
