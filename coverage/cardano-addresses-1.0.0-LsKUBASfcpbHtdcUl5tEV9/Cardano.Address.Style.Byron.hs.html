   91 -- $overview
   92 --
   93 -- This module provides an implementation of:
   94 --
   95 -- - 'GenMasterKey': for generating Byron master keys from mnemonic sentences
   96 -- - 'HardDerivation': for hierarchical derivation of parent to child keys
   97 -- - 'PaymentAddress': for constructing addresses from a public key
   98 --
   99 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  100 -- days of Cardano. Using this type of addresses and underlying key scheme is
  101 -- now considered __deprecated__ because of some security implications.
  102 --
  103 -- Unless you have good reason to do so (like writing backward-compatible code
  104 -- with an existing piece), any new implementation __should use__ the 'Icarus'
  105 -- style for key and addresses.
  106 --
  107 -- The internals of the 'Byron' does not matter for the reader, but basically
  108 -- contains what is necessary to perform key derivation and generate addresses
  109 -- from a 'Byron' type.
  110 
  111 {-------------------------------------------------------------------------------
  112                                    Key Types
  113 -------------------------------------------------------------------------------}

  115 -- | Material for deriving HD random scheme keys, which can be used for making
  116 -- addresses.
  117 --
  118 -- @since 1.0.0
  119 data Byron (depth :: Depth) key = Byron
  120     { getKey :: key
  121     -- ^ The raw private or public key.
  122     --
  123     -- @since 1.0.0
  124     , derivationPath :: DerivationPath depth
  125     -- ^ The address derivation indices for the level of this key.
  126     --
  127     -- @since 1.0.0
  128     , payloadPassphrase :: ScrubbedBytes
  129     -- ^ Used for encryption of the derivation path payload within an address.
  130     --
  131     -- @since 1.0.0
  132     } deriving stock (Generic)
  133 {-# DEPRECATED Byron "see Cardano.Address.Style.Icarus" #-}
  134 {-# DEPRECATED getKey "see Cardano.Address.Style.Icarus" #-}
  135 {-# DEPRECATED derivationPath "see Cardano.Address.Style.Icarus" #-}
  136 {-# DEPRECATED payloadPassphrase "see Cardano.Address.Style.Icarus" #-}
  137 
  138 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
  139 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
  140 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
  141 deriving instance (Functor (Byron depth))

  169 -- | Backdoor for generating a new key from a raw XPrv.
  170 --
  171 -- Note that the @depth@ is left open so that the caller gets to decide what type
  172 -- of key this is. This is mostly for testing, in practice, seeds are used to
  173 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  174 --
  175 -- The first argument is a type-family 'DerivationPath' and its type depends on
  176 -- the 'depth' of the key.
  177 --
  178 -- __examples:__
  179 --
  180 -- >>> liftXPrv () prv
  181 -- _ :: Byron RootK XPrv
  182 --
  183 -- >>> liftXPrv minBound prv
  184 -- _ :: Byron AccountK XPrv
  185 --
  186 -- >>> liftXPrv (minBound, minBound) prv
  187 -- _ :: Byron AddressK XPrv
  188 --
  189 -- @since 1.0.0
  190 liftXPrv
  191     :: DerivationPath depth
  192     -> XPrv
  193     -> Byron depth XPrv
  194 liftXPrv derivationPath getKey = Byron
  195     { getKey
  196     , derivationPath
  197     , payloadPassphrase = hdPassphrase (toXPub getKey)
  198     }
  199 {-# DEPRECATED liftXPrv "see Cardano.Address.Style.Icarus" #-}
  200 
  201 -- | The hierarchical derivation indices for a given level/depth.
  202 --
  203 -- @since 1.0.0
  204 type family DerivationPath (depth :: Depth) :: * where
  205     -- The root key is generated from the seed.
  206     DerivationPath 'RootK =
  207         ()
  208     -- The account key is generated from the root key and account index.
  209     DerivationPath 'AccountK =
  210         Index 'WholeDomain 'AccountK
  211     -- The address key is generated from the account key and address index.
  212     DerivationPath 'AddressK =
  213         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  214 {-# DEPRECATED DerivationPath "see Cardano.Address.Style.Icarus" #-}

  216 instance GenMasterKey Byron where
  217     type SecondFactor Byron = ()
  218 
  219     genMasterKeyFromXPrv = liftXPrv ()
  220     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  221         liftXPrv () xprv
  222       where
  223         xprv = generate (hashSeed seedValidated)
  224         seed  = entropyToBytes $ mnemonicToEntropy mw
  225         seedValidated = assert
  226             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  227             seed
  228 
  229 instance HardDerivation Byron where
  230     type AddressIndexDerivationType Byron = 'WholeDomain
  231     type AccountIndexDerivationType Byron = 'WholeDomain
  232     type WithAccountStyle Byron = ()
  233 
  234     deriveAccountPrivateKey rootXPrv accIx = Byron
  235         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  236         , derivationPath = accIx
  237         , payloadPassphrase = payloadPassphrase rootXPrv
  238         }
  239 
  240     deriveAddressPrivateKey accXPrv () addrIx = Byron
  241         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  242         , derivationPath = (derivationPath accXPrv, addrIx)
  243         , payloadPassphrase = payloadPassphrase accXPrv
  244         }

  246 instance HasNetworkDiscriminant Byron where
  247     type NetworkDiscriminant Byron = (AddressDiscrimination, NetworkTag)
  248     addressDiscrimination = fst
  249     networkTag = snd

  251 instance PaymentAddress Byron where
  252     paymentAddress discrimination k = unsafeMkAddress
  253         $ CBOR.toStrictByteString
  254         $ CBOR.encodeAddress (getKey k) attrs
  255       where
  256         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  257         pwd = payloadPassphrase k
  258         NetworkTag magic = networkTag @Byron discrimination
  259         attrs = case addressDiscrimination @Byron discrimination of
  260             RequiresNetworkTag ->
  261                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  262                 , CBOR.encodeProtocolMagicAttr magic
  263                 ]
  264             RequiresNoTag ->
  265                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  266                 ]

  272 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  273 minSeedLengthBytes :: Int
  274 minSeedLengthBytes = 16
  275 
  276 -- | Hash the seed entropy (generated from mnemonic) used to initiate a HD
  277 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  278 -- minimum for 'generate' (32 bytes).
  279 --
  280 -- Note that our current implementation deviates from BIP-39 because we use a
  281 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  282 --
  283 -- There are two methods of hashing the seed entropy, for different use cases.
  284 --
  285 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  286 --    Blake2b_256, inside a double CBOR serialization sandwich.
  287 --
  288 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  289 --    Blake2b_256, without any serialization.
  290 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  291 hashSeed = serialize . blake2b256 . serialize
  292   where
  293     serialize = BA.convert . cbor . BA.convert
  294     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  295 
  296 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  297 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  298 
  299 -- | Derive a symmetric key for encrypting and authenticating the address
  300 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  301 -- is employed.
  302 hdPassphrase :: XPub -> ScrubbedBytes
  303 hdPassphrase masterKey =
  304     PBKDF2.generate
  305     (PBKDF2.prfHMAC SHA512)
  306     (PBKDF2.Parameters 500 32)
  307     (xpubToBytes masterKey)
  308     ("address-hashing" :: ByteString)
