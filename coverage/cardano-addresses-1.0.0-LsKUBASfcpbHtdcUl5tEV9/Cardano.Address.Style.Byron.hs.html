  100 -- $overview
  101 --
  102 -- This module provides an implementation of:
  103 --
  104 -- - 'Cardano.Address.Derivation.GenMasterKey': for generating Byron master keys from mnemonic sentences
  105 -- - 'Cardano.Address.Derivation.HardDerivation': for hierarchical derivation of parent to child keys
  106 -- - 'Cardano.Address.PaymentAddress': for constructing addresses from a public key
  107 --
  108 -- We call 'Byron' addresses the old address type used by Daedalus in the early
  109 -- days of Cardano. Using this type of addresses and underlying key scheme is
  110 -- now considered __deprecated__ because of some security implications.
  111 --
  112 -- The internals of the 'Byron' does not matter for the reader, but basically
  113 -- contains what is necessary to perform key derivation and generate addresses
  114 -- from a 'Byron' type.
  115 --
  116 -- == Deprecation Notice
  117 --
  118 -- Unless you have good reason to do so (like writing backward-compatible code
  119 -- with an existing piece), any new implementation __should use__ the
  120 -- 'Cardano.Address.Style.Icarus.Icarus' style for key and addresses.
  121 
  122 
  123 -- | Material for deriving HD random scheme keys, which can be used for making
  124 -- addresses.
  125 --
  126 -- @since 1.0.0
  127 data Byron (depth :: Depth) key = Byron
  128     { getKey :: key
  129     -- ^ The raw private or public key.
  130     --
  131     -- @since 1.0.0
  132     , derivationPath :: DerivationPath depth
  133     -- ^ The address derivation indices for the level of this key.
  134     --
  135     -- @since 1.0.0
  136     , payloadPassphrase :: ScrubbedBytes
  137     -- ^ Used for encryption of the derivation path payload within an address.
  138     --
  139     -- @since 1.0.0
  140     } deriving stock (Generic)
  141 {-# DEPRECATED Byron "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  142 {-# DEPRECATED getKey "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  143 {-# DEPRECATED derivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}
  144 {-# DEPRECATED payloadPassphrase "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  146 instance (NFData key, NFData (DerivationPath depth)) => NFData (Byron depth key)
  147 deriving instance (Show key, Show (DerivationPath depth)) => Show (Byron depth key)
  148 deriving instance (Eq key, Eq (DerivationPath depth)) => Eq (Byron depth key)
  149 deriving instance (Functor (Byron depth))

  151 -- | The hierarchical derivation indices for a given level/depth.
  152 --
  153 -- @since 1.0.0
  154 type family DerivationPath (depth :: Depth) :: * where
  155     -- The root key is generated from the seed.
  156     DerivationPath 'RootK =
  157         ()
  158     -- The account key is generated from the root key and account index.
  159     DerivationPath 'AccountK =
  160         Index 'WholeDomain 'AccountK
  161     -- The address key is generated from the account key and address index.
  162     DerivationPath 'AddressK =
  163         (Index 'WholeDomain 'AccountK, Index 'WholeDomain 'AddressK)
  164 {-# DEPRECATED DerivationPath "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  166 --
  167 -- Key Derivation
  168 --
  169 -- === Generating a root key from 'SomeMnemonic'
  170 -- > :set -XOverloadedStrings
  171 -- > :set -XTypeApplications
  172 -- > :set -XDataKinds
  173 -- > import Cardano.Mnemonic ( mkSomeMnemonic )
  174 -- >
  175 -- > let (Right mw) = mkSomeMnemonic @'[12] ["moon","fox","ostrich","quick","cactus","raven","wasp","intact","first","ring","crumble","error"]
  176 -- > let rootK = genMasterKeyFromMnemonic mw :: Byron 'RootK XPrv
  177 --
  178 -- === Deriving child keys
  179 --
  180 -- > let accIx = toEnum 0x80000000
  181 -- > let acctK = deriveAccountPrivateKey rootK accIx
  182 -- >
  183 -- > let addIx = toEnum 0x80000014
  184 -- > let addrK = deriveAddressPrivateKey acctK addIx
  185 
  186 instance Internal.GenMasterKey Byron where
  187     type SecondFactor Byron = ()
  188 
  189     genMasterKeyFromXPrv = liftXPrv ()
  190     genMasterKeyFromMnemonic (SomeMnemonic mw) () =
  191         liftXPrv () xprv
  192       where
  193         xprv = generate (hashSeed seedValidated)
  194         seed  = entropyToBytes $ mnemonicToEntropy mw
  195         seedValidated = assert
  196             (BA.length seed >= minSeedLengthBytes && BA.length seed <= 255)
  197             seed
  198 
  199 instance Internal.HardDerivation Byron where
  200     type AddressIndexDerivationType Byron = 'WholeDomain
  201     type AccountIndexDerivationType Byron = 'WholeDomain
  202     type WithAccountStyle Byron = ()
  203 
  204     deriveAccountPrivateKey rootXPrv accIx = Byron
  205         { getKey = deriveXPrv DerivationScheme1 (getKey rootXPrv) accIx
  206         , derivationPath = accIx
  207         , payloadPassphrase = payloadPassphrase rootXPrv
  208         }
  209 
  210     deriveAddressPrivateKey accXPrv () addrIx = Byron
  211         { getKey = deriveXPrv DerivationScheme1 (getKey accXPrv) addrIx
  212         , derivationPath = (derivationPath accXPrv, addrIx)
  213         , payloadPassphrase = payloadPassphrase accXPrv
  214         }

  275 instance Internal.PaymentAddress Byron where
  276     paymentAddress discrimination k = unsafeMkAddress
  277         $ CBOR.toStrictByteString
  278         $ CBOR.encodeAddress (getKey k) attrs
  279       where
  280         (acctIx, addrIx) = bimap word32 word32 $ derivationPath k
  281         pwd = payloadPassphrase k
  282         NetworkTag magic = networkTag @Byron discrimination
  283         attrs = case addressDiscrimination @Byron discrimination of
  284             RequiresNetworkTag ->
  285                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  286                 , CBOR.encodeProtocolMagicAttr magic
  287                 ]
  288             RequiresNoTag ->
  289                 [ CBOR.encodeDerivationPathAttr pwd acctIx addrIx
  290                 ]

  336 -- | Backdoor for generating a new key from a raw XPrv.
  337 --
  338 -- Note that the @depth@ is left open so that the caller gets to decide what type
  339 -- of key this is. This is mostly for testing, in practice, seeds are used to
  340 -- represent root keys, and one should 'genMasterKeyFromXPrv'
  341 --
  342 -- The first argument is a type-family 'DerivationPath' and its type depends on
  343 -- the 'depth' of the key.
  344 --
  345 -- __examples:__
  346 --
  347 -- >>> liftXPrv () prv
  348 -- _ :: Byron RootK XPrv
  349 --
  350 -- >>> liftXPrv minBound prv
  351 -- _ :: Byron AccountK XPrv
  352 --
  353 -- >>> liftXPrv (minBound, minBound) prv
  354 -- _ :: Byron AddressK XPrv
  355 --
  356 -- @since 1.0.0
  357 liftXPrv
  358     :: DerivationPath depth
  359     -> XPrv
  360     -> Byron depth XPrv
  361 liftXPrv derivationPath getKey = Byron
  362     { getKey
  363     , derivationPath
  364     , payloadPassphrase = hdPassphrase (toXPub getKey)
  365     }
  366 {-# DEPRECATED liftXPrv "see 'Cardano.Address.Style.Icarus.Icarus'" #-}

  372 -- The amount of entropy carried by a BIP-39 12-word mnemonic is 16 bytes.
  373 minSeedLengthBytes :: Int
  374 minSeedLengthBytes = 16
  375 
  376 -- Hash the seed entropy (generated from mnemonic) used to initiate a HD
  377 -- wallet. This increases the key length to 34 bytes, selectKey is greater than the
  378 -- minimum for 'generate' (32 bytes).
  379 --
  380 -- Note that our current implementation deviates from BIP-39 because we use a
  381 -- hash function (Blake2b) rather than key stretching with PBKDF2.
  382 --
  383 -- There are two methods of hashing the seed entropy, for different use cases.
  384 --
  385 -- 1. Normal random derivation wallet seeds. The seed entropy is hashed using
  386 --    Blake2b_256, inside a double CBOR serialization sandwich.
  387 --
  388 -- 2. Seeds for redeeming paper wallets. The seed entropy is hashed using
  389 --    Blake2b_256, without any serialization.
  390 hashSeed :: ScrubbedBytes -> ScrubbedBytes
  391 hashSeed = serialize . blake2b256 . serialize
  392   where
  393     serialize = BA.convert . cbor . BA.convert
  394     cbor = CBOR.toStrictByteString . CBOR.encodeBytes
  395 
  396 -- Hash a byte string through blake2b 256
  397 blake2b256 :: ScrubbedBytes -> ScrubbedBytes
  398 blake2b256 = BA.convert . hash @ScrubbedBytes @Blake2b_256
  399 
  400 -- Derive a symmetric key for encrypting and authenticating the address
  401 -- derivation path. PBKDF2 encryption using HMAC with the hash algorithm SHA512
  402 -- is employed.
  403 hdPassphrase :: XPub -> ScrubbedBytes
  404 hdPassphrase masterKey =
  405     PBKDF2.generate
  406     (PBKDF2.prfHMAC SHA512)
  407     (PBKDF2.Parameters 500 32)
  408     (xpubToBytes masterKey)
  409     ("address-hashing" :: ByteString)

  411 -- Relatively unsafe conversion of an enum to 'Word32'
  412 word32 :: Enum a => a -> Word32
  413 word32 = fromIntegral . fromEnum
