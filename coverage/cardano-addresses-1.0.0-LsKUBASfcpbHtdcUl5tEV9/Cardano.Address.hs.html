   62 -- | An 'Address' type representing 'Cardano' addresses. Internals are
   63 -- irrevelant to the user.
   64 --
   65 -- @since 1.0.0
   66 newtype Address = Address
   67     { unAddress :: ByteString
   68     } deriving stock (Generic, Show, Eq, Ord)
   69 instance NFData Address
   70 
   71 -- Unsafe constructor for easily lifting bytes inside an 'Address'.
   72 --
   73 -- /!\ Use at your own risks.
   74 unsafeMkAddress :: ByteString -> Address
   75 unsafeMkAddress = Address
   76 
   77 -- | Encode an 'Address' to a base58 'Text'.
   78 --
   79 -- @since 1.0.0
   80 base58 :: Address -> Text
   81 base58 = T.decodeUtf8 . encodeBase58 bitcoinAlphabet . unAddress
   82 
   83 -- | Decode a base58-encoded 'Text' into an 'Address'
   84 --
   85 -- @since 1.0.0
   86 fromBase58 :: Text -> Maybe Address
   87 fromBase58 =
   88     deserialiseCbor (unsafeMkAddress <$> decodeAddress)
   89    <=<
   90     decodeBase58 bitcoinAlphabet . T.encodeUtf8
   91 
   92 -- | Encode an 'Address' to bech32 'Text', using @addr@ as a human readable prefix.
   93 --
   94 -- @since 1.0.0
   95 bech32 :: Address -> Text
   96 bech32 = Bech32.encodeLenient hrp . Bech32.dataPartFromBytes . unAddress
   97   where
   98     hrp = [Bech32.humanReadablePart|addr|]
   99 
  100 -- | Decode a bech32-encoded  'Text' into an 'Address'
  101 --
  102 -- @since 1.0.0
  103 fromBech32 :: Text -> Maybe Address
  104 fromBech32 =
  105     fmap unsafeMkAddress . Bech32.dataPartToBytes
  106    <=<
  107     either (const Nothing) (Just . snd) . Bech32.decodeLenient

  144 -- Magic constant associated with a given network. This is mainly used in two
  145 -- places:
  146 --
  147 -- (1) In 'Address' payloads, to discriminate addresses between networks.
  148 -- (2) At the network-level, when doing handshake with nodes.
  149 newtype NetworkTag
  150     = NetworkTag Word32
  151     deriving (Generic, Show, Eq)
  152 instance NFData NetworkTag
  153 
  154 -- Describe requirements for address discrimination on the Byron era.
  155 data AddressDiscrimination
  156     = RequiresNetworkTag
  157     | RequiresNoTag
  158     deriving (Generic, Show, Eq)
  159 instance NFData AddressDiscrimination

  161 invariantSize :: HasCallStack => Int -> ByteString -> ByteString
  162 invariantSize expectedLength bytes
  163     | BS.length bytes == expectedLength = bytes
  164     | otherwise = error
  165       $ "length was "
  166       ++ show (BS.length bytes)
  167       ++ ", but expected to be "
  168       ++ (show expectedLength)
  169 
  170 invariantNetworkTag :: HasCallStack => Word32 -> NetworkTag -> Word8
  171 invariantNetworkTag limit (NetworkTag num)
  172     | num < limit = fromIntegral num
  173     | otherwise = error
  174       $ "network tag was "
  175       ++ show num
  176       ++ ", but expected to be less than "
  177       ++ show limit
